19/// This class keeps offsets to the debug sections. Any object which is
20/// supposed to be emitted into the debug section should use this class to
21/// track debug sections offsets.
22class OutputSections {
23public:
24 /// List of tracked debug sections.
25 enum class DebugSectionKind : uint8_t {
26 DebugInfo = 0,
27 DebugLine,
28 DebugFrame,
29 DebugRange,
30 DebugRngLists,
31 DebugLoc,
32 DebugLocLists,
33 DebugARanges,
34 DebugAbbrev,
35 DebugMacinfo,
36 DebugMacro,
37 };
38 constexpr static size_t SectionKindsNum = 11;
39
40 /// Recognise the section name and match it with the DebugSectionKind.
41 static std::optional<DebugSectionKind> parseDebugSectionName(StringRef Name);
42
43 /// When objects(f.e. compile units) are glued into the single file,
44 /// the debug sections corresponding to the concrete object are assigned
45 /// with offsets inside the whole file. This method returns offset
46 /// to the \p SectionKind debug section, corresponding to this object.
47 uint64_t getStartOffset(DebugSectionKind SectionKind) const {
48 return Offsets[static_cast<
49 typename std::underlying_type<DebugSectionKind>::type>(SectionKind)];
50 }
51
52 /// Set offset to the start of specified \p SectionKind debug section,
53 /// corresponding to this object.
54 void setStartOffset(DebugSectionKind SectionKind, uint64_t Offset) {
55 Offsets[static_cast<typename std::underlying_type<DebugSectionKind>::type>(
56 SectionKind)] = Offset;
57 }
58
59protected:
60 /// Offsets to the debug sections composing this object.
61 std::array<uint64_t, SectionKindsNum> Offsets = {0};
62};
