23/// Stores all information related to a compile unit, be it in its original
24/// instance of the object file or its brand new cloned and generated DIE tree.
25class CompileUnit : public DwarfUnit {
26public:
27 CompileUnit(LinkContext &, unsigned ID, StringRef ClangModuleName,
28 DWARFFile &File,
29 DWARFLinker::SwiftInterfacesMapTy *,
30 UnitMessageHandlerTy WarningHandler)
31 : DwarfUnit(ID, ClangModuleName, WarningHandler), ContaingFile(File) {
32 FormParams.Version = 4;
33 FormParams.Format = dwarf::DWARF32;
34 FormParams.AddrSize = 4;
35 UnitName = ContaingFile.FileName;
36 }
37
38 CompileUnit(LinkContext &, DWARFUnit &OrigUnit, unsigned ID,
39 StringRef ClangModuleName, DWARFFile &File,
40 UnitMessageHandlerTy WarningHandler)
41 : DwarfUnit(ID, ClangModuleName, WarningHandler),
42 ContaingFile(File), OrigUnit(&OrigUnit) {
43 DWARFDie CUDie = OrigUnit.getUnitDIE();
44 if (!CUDie)
45 return;
46
47 if (File.Dwarf)
48 Endianess = File.Dwarf->isLittleEndian() ? support::endianness::little
49 : support::endianness::big;
50
51 FormParams.Version = OrigUnit.getVersion();
52 FormParams.Format = dwarf::DWARF32;
53 FormParams.AddrSize = OrigUnit.getAddressByteSize();
54
55 Language = dwarf::toUnsigned(CUDie.find(dwarf::DW_AT_language), 0);
56
57 UnitName = ContaingFile.FileName;
58 SysRoot = dwarf::toStringRef(CUDie.find(dwarf::DW_AT_LLVM_sysroot)).str();
59 }

124 std::optional<DWARFFormValue> find(const DWARFDebugInfoEntry *Die,
125 ArrayRef<dwarf::Attribute> Attrs) const {
126 if (!Die)
127 return std::nullopt;
128 auto AbbrevDecl = Die->getAbbreviationDeclarationPtr();
129 if (AbbrevDecl) {
130 for (auto Attr : Attrs) {
131 if (auto Value = AbbrevDecl->getAttributeValue(Die->getOffset(), Attr,
132 *OrigUnit))
133 return Value;
134 }
135 }
136 return std::nullopt;
137 }
