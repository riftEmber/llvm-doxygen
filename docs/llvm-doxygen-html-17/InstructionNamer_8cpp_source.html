23namespace {
24void nameInstructions(Function &F) {
25 for (auto &Arg : F.args()) {
26 if (!Arg.hasName())
27 Arg.setName("arg");
28 }
29
30 for (BasicBlock &BB : F) {
31 if (!BB.hasName())
32 BB.setName("bb");
33
34 for (Instruction &I : BB) {
35 if (!I.hasName() && !I.getType()->isVoidTy())
36 I.setName("i");
37 }
38 }
39}
