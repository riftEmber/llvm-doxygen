170 /// Keeps track of data associated with one object during linking.
171 /// i.e. source file descriptor, compilation units, output data
172 /// for compilation units common tables.
173 struct LinkContext : public OutputSections {
174 using UnitListTy = SmallVector<std::unique_ptr<CompileUnit>>;
175
176 /// Keep information for referenced clang module: already loaded DWARF info
177 /// of the clang module and a CompileUnit of the module.
178 struct RefModuleUnit {
179 RefModuleUnit(DWARFFile &File, std::unique_ptr<CompileUnit> Unit)
180 : File(File), Unit(std::move(Unit)) {}
181 RefModuleUnit(RefModuleUnit &&Other)
182 : File(Other.File), Unit(std::move(Other.Unit)) {}
183 RefModuleUnit(const RefModuleUnit &) = delete;
184
185 DWARFFile &File;
186 std::unique_ptr<CompileUnit> Unit;
187 };
188 using ModuleUnitListTy = SmallVector<RefModuleUnit>;
189
190 /// Object file descriptor.
191 DWARFFile &File;
192
193 /// Set of Compilation Units(may be accessed asynchroniously for reading).
194 UnitListTy CompileUnits;
195
196 /// Set of Compile Units for modules.
197 ModuleUnitListTy ModulesCompileUnits;
198
199 /// Size of Debug info before optimizing.
200 uint64_t OriginalDebugInfoSize = 0;
201
202 /// Output sections, common for all compilation units.
203 OutTablesFileTy OutDebugInfoBytes;
204
205 /// Endianness for the final file.
206 support::endianness Endianess = support::endianness::little;
207
208 LinkContext(DWARFFile &File) : File(File) {
209 if (File.Dwarf) {
210 if (!File.Dwarf->compile_units().empty())
211 CompileUnits.reserve(File.Dwarf->getNumCompileUnits());
212
213 Endianess = File.Dwarf->isLittleEndian() ? support::endianness::little
214 : support::endianness::big;
215 }
216 }
217
218 /// Add Compile Unit corresponding to the module.
219 void addModulesCompileUnit(RefModuleUnit &&Unit) {
220 ModulesCompileUnits.emplace_back(std::move(Unit));
221 }
222
223 /// Return Endiannes of the source DWARF information.
224 support::endianness getEndianness() { return Endianess; }
225
226 /// \returns pointer to compilation unit which corresponds \p Offset.
227 CompileUnit *getUnitForOffset(CompileUnit &CU, uint64_t Offset) const;
228 };
