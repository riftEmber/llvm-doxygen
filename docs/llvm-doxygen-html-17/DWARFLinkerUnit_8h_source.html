22using UnitMessageHandlerTy = function_ref<void(
23 const Twine &Error, StringRef Context, const DWARFDie *DIE)>;

25/// Each unit keeps output data as a file with debug tables
26/// corresponding to the concrete unit.
27using OutTablesFileTy = SmallString<0>;
28
29/// Base class for all Dwarf units(Compile unit/Type table unit).
30class DwarfUnit : public OutputSections {
31public:
32 virtual ~DwarfUnit() {}
33 DwarfUnit(unsigned ID, StringRef ClangModuleName,
34 UnitMessageHandlerTy WarningHandler)
35 : ID(ID), ClangModuleName(ClangModuleName),
36 WarningHandler(WarningHandler) {
37 FormParams.Version = 4;
38 FormParams.Format = dwarf::DWARF32;
39 FormParams.AddrSize = 4;
40 }
41
42 /// Endiannes for the compile unit.
43 support::endianness getEndianness() const { return Endianess; }
44
45 /// Return DWARF version.
46 uint16_t getVersion() const { return FormParams.Version; }
47
48 /// Return size of header of debug_info table.
49 uint16_t getHeaderSize() const { return FormParams.Version >= 5 ? 12 : 11; }
50
51 /// Return size of address.
52 uint8_t getAddressByteSize() const { return FormParams.AddrSize; }
53
54 /// Return size of reference.
55 uint8_t getRefAddrByteSize() const { return FormParams.getRefAddrByteSize(); }
56
57 /// Return format of the Dwarf(DWARF32 or DWARF64).
58 /// TODO: DWARF64 is not currently supported.
59 dwarf::DwarfFormat getDwarfFormat() const { return FormParams.Format; }
60
61 /// Unique id of the unit.
62 unsigned getUniqueID() const { return ID; }
63
64 /// Return language of this unit.
65 uint16_t getLanguage() const { return Language; }
66
67 /// Set size of this(newly generated) compile unit.
68 void setUnitSize(uint64_t UnitSize) { this->UnitSize = UnitSize; }
69
70 /// Returns size of this(newly generated) compile unit.
71 uint64_t getUnitSize() const { return UnitSize; }
72
73 /// Returns this unit name.
74 StringRef getUnitName() const { return UnitName; }
75
76 /// Return the DW_AT_LLVM_sysroot of the compile unit or an empty StringRef.
77 StringRef getSysRoot() { return SysRoot; }
78
79 /// Create a Die for this unit.
80 void setOutputDIE(DIE *UnitDie) { NewUnit = UnitDie; }
81
82 /// Return Die for this compile unit.
83 DIE *getOutputUnitDIE() const { return NewUnit; }
84
85 /// Return true if this compile unit is from Clang module.
86 bool isClangModule() const { return !ClangModuleName.empty(); }
87
88 /// Return Clang module name;
89 const std::string &getClangModuleName() const { return ClangModuleName; }
90
91 /// Returns generated file keeping debug tables for this compile unit.
92 OutTablesFileTy &getOutDwarfBits() { return OutDebugInfoBits; }
93
94 /// Erases generated file keeping debug tables for this compile unit.
95 void eraseDwarfBits() { OutDebugInfoBits = OutTablesFileTy(); }
96
97 MCSymbol *getLabelBegin() { return LabelBegin; }
98 void setLabelBegin(MCSymbol *S) { LabelBegin = S; }
99
100 /// Error reporting methods.
101 /// @{
102
103 void reportWarning(const Twine &Warning,
104 const DWARFDie *Die = nullptr) const {
105 if (WarningHandler)
106 WarningHandler(Warning, getUnitName(), Die);
107 }
108 void reportWarning(Error Warning) const {
109 handleAllErrors(std::move(Warning), [&](ErrorInfoBase &Info) {
110 if (WarningHandler)
111 WarningHandler(Info.message(), getUnitName(), nullptr);
112 });
113 }
114 /// @}
115
116 /// This structure keeps fields which would be used for creating accelerator
117 /// table.
118 struct AccelInfo {
119 AccelInfo(StringEntry *Name, const DIE *Die, bool SkipPubSection = false);
120 AccelInfo(StringEntry *Name, const DIE *Die, uint32_t QualifiedNameHash,
121 bool ObjCClassIsImplementation);
122
123 /// Name of the entry.
124 StringEntry *Name = nullptr;
125
126 /// Tag of the DIE this entry describes.
127 dwarf::Tag Tag = dwarf::DW_TAG_null;
128
129 /// Output offset of the DIE this entry describes.
130 uint64_t OutOffset = 0;
131
132 /// Hash of the fully qualified name.
133 uint32_t QualifiedNameHash = 0;
134
135 /// Emit this entry only in the apple_* sections.
136 bool SkipPubSection = false;
137
138 /// Is this an ObjC class implementation?
139 bool ObjcClassImplementation = false;
140
141 /// Cloned Die containing acceleration info.
142 const DIE *Die = nullptr;
143 };
144
145protected:
146 /// Unique ID for the unit.
147 unsigned ID = 0;
148
149 /// Properties of the unit.
150 dwarf::FormParams FormParams;
151
152 /// DIE for newly generated compile unit.
153 DIE *NewUnit = nullptr;
154
155 /// The DW_AT_language of this unit.
156 uint16_t Language = 0;
157
158 /// The name of this unit.
159 std::string UnitName;
160
161 /// The DW_AT_LLVM_sysroot of this unit.
162 std::string SysRoot;
163
164 /// If this is a Clang module, this holds the module's name.
165 std::string ClangModuleName;
166
167 uint64_t UnitSize = 0;
168
169 /// Elf file containg generated debug tables for this compile unit.
170 OutTablesFileTy OutDebugInfoBits;
171
172 /// Endiannes for this compile unit.
173 support::endianness Endianess = support::endianness::little;
174
175 MCSymbol *LabelBegin = nullptr;
176
177 /// true if current unit references_to/is_referenced by other unit.
178 std::atomic<bool> IsInterconnectedCU = {false};
179
180 UnitMessageHandlerTy WarningHandler;
181};
