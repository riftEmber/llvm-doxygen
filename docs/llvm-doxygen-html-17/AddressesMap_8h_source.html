21/// linked address.
22using RangesTy = AddressRangesMap;

24/// AddressesMap represents information about valid addresses used
25/// by debug information. Valid addresses are those which points to
26/// live code sections. i.e. relocations for these addresses point
27/// into sections which would be/are placed into resulting binary.
28class AddressesMap {
29public:
30 virtual ~AddressesMap() = default;
31
32 /// Checks that there are valid relocations in the .debug_info
33 /// section.
34 virtual bool hasValidRelocs() = 0;
35
36 /// Checks that the specified DWARF expression operand \p Op references live
37 /// code section and returns the relocation adjustment value (to get the
38 /// linked address this value might be added to the source expression operand
39 /// address).
40 /// \returns relocation adjustment value or std::nullopt if there is no
41 /// corresponding live address.
42 virtual std::optional<int64_t>
43 getExprOpAddressRelocAdjustment(DWARFUnit &U,
44 const DWARFExpression::Operation &Op,
45 uint64_t StartOffset, uint64_t EndOffset) = 0;
46
47 /// Checks that the specified subprogram \p DIE references the live code
48 /// section and returns the relocation adjustment value (to get the linked
49 /// address this value might be added to the source subprogram address).
50 /// Allowed kinds of input DIE: DW_TAG_subprogram, DW_TAG_label.
51 /// \returns relocation adjustment value or std::nullopt if there is no
52 /// corresponding live address.
53 virtual std::optional<int64_t>
54 getSubprogramRelocAdjustment(const DWARFDie &DIE) = 0;
55
56 /// Apply the valid relocations to the buffer \p Data, taking into
57 /// account that Data is at \p BaseOffset in the .debug_info section.
58 ///
59 /// \returns true whether any reloc has been applied.
60 virtual bool applyValidRelocs(MutableArrayRef<char> Data, uint64_t BaseOffset,
61 bool IsLittleEndian) = 0;
62
63 /// Erases all data.
64 virtual void clear() = 0;
65};
