37ArchitectureSet::operator std::string() const {
38 if (empty())
39 return "[(empty)]";
40
41 std::string result;
42 auto size = count();
43 for (auto arch : *this) {
44 result.append(std::string(getArchitectureName(arch)));
45 size -= 1;
46 if (size)
47 result.append(" ");
48 }
49 return result;
