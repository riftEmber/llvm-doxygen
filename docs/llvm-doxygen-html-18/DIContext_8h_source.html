31/// A format-neutral container for source line information.
32struct DILineInfo {
33 // DILineInfo contains "<invalid>" for function/filename it cannot fetch.
34 static constexpr const char *const BadString = "<invalid>";
35 // Use "??" instead of "<invalid>" to make our output closer to addr2line.
36 static constexpr const char *const Addr2LineBadString = "??";
37 std::string FileName;
38 std::string FunctionName;
39 std::string StartFileName;
40 // Full source corresponding to `FileName`
41 std::optional<StringRef> Source;
42 // Source code for this particular line
43 // (in case if `Source` is not available)
44 std::optional<StringRef> LineSource;
45 uint32_t Line = 0;
46 uint32_t Column = 0;
47 uint32_t StartLine = 0;
48 std::optional<uint64_t> StartAddress;
49
50 // DWARF-specific.
51 uint32_t Discriminator = 0;
52
53 DILineInfo()
54 : FileName(BadString), FunctionName(BadString), StartFileName(BadString) {
55 }
56
57 bool operator==(const DILineInfo &RHS) const {
58 return Line == RHS.Line && Column == RHS.Column &&
59 FileName == RHS.FileName && FunctionName == RHS.FunctionName &&
60 StartFileName == RHS.StartFileName && StartLine == RHS.StartLine &&
61 Discriminator == RHS.Discriminator;
62 }
63
64 bool operator!=(const DILineInfo &RHS) const { return !(*this == RHS); }
65
66 bool operator<(const DILineInfo &RHS) const {
67 return std::tie(FileName, FunctionName, StartFileName, Line, Column,
68 StartLine, Discriminator) <
69 std::tie(RHS.FileName, RHS.FunctionName, RHS.StartFileName, RHS.Line,
70 RHS.Column, RHS.StartLine, RHS.Discriminator);
71 }
72
73 explicit operator bool() const { return *this != DILineInfo(); }
74
75 void dump(raw_ostream &OS) {
76 OS << "Line info: ";
77 if (FileName != BadString)
78 OS << "file '" << FileName << "', ";
79 if (FunctionName != BadString)
80 OS << "function '" << FunctionName << "', ";
81 OS << "line " << Line << ", ";
82 OS << "column " << Column << ", ";
83 if (StartFileName != BadString)
84 OS << "start file '" << StartFileName << "', ";
85 OS << "start line " << StartLine << '\n';
86 }
87};

95public:

98 /// Returns the frame at `Index`. Frames are stored in bottom-up
99 /// (leaf-to-root) order with increasing index.
100 const DILineInfo &getFrame(unsigned Index) const {
101 assert(Index < Frames.size());
102 return Frames[Index];
103 }
104
105 DILineInfo *getMutableFrame(unsigned Index) {
106 assert(Index < Frames.size());
107 return &Frames[Index];
108 }
109

111
112 void addFrame(const DILineInfo &Frame) { Frames.push_back(Frame); }

269/// An inferface for inquiring the load address of a loaded object file
270/// to be used by the DIContext implementations when applying relocations
271/// on the fly.
272class LoadedObjectInfo {
273protected:
274 LoadedObjectInfo() = default;
275 LoadedObjectInfo(const LoadedObjectInfo &) = default;
276
277public:
278 virtual ~LoadedObjectInfo() = default;
279
280 /// Obtain the Load Address of a section by SectionRef.
281 ///
282 /// Calculate the address of the given section.
283 /// The section need not be present in the local address space. The addresses
284 /// need to be consistent with the addresses used to query the DIContext and
285 /// the output of this function should be deterministic, i.e. repeated calls
286 /// with the same Sec should give the same address.
287 virtual uint64_t getSectionLoadAddress(const object::SectionRef &Sec) const {
288 return 0;
289 }
290
291 /// If conveniently available, return the content of the given Section.
292 ///
293 /// When the section is available in the local address space, in relocated
294 /// (loaded) form, e.g. because it was relocated by a JIT for execution, this
295 /// function should provide the contents of said section in `Data`. If the
296 /// loaded section is not available, or the cost of retrieving it would be
297 /// prohibitive, this function should return false. In that case, relocations
298 /// will be read from the local (unrelocated) object file and applied on the
299 /// fly. Note that this method is used purely for optimzation purposes in the
300 /// common case of JITting in the local address space, so returning false
301 /// should always be correct.
302 virtual bool getLoadedSectionContents(const object::SectionRef &Sec,
303 StringRef &Data) const {
304 return false;
305 }
306
307 // FIXME: This is untested and unused anywhere in the LLVM project, it's
308 // used/needed by Julia (an external project). It should have some coverage
309 // (at least tests, but ideally example functionality).
310 /// Obtain a copy of this LoadedObjectInfo.
311 virtual std::unique_ptr<LoadedObjectInfo> clone() const = 0;
312};
