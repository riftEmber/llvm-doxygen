17Error DebugFrameDataSubsectionRef::initialize(BinaryStreamReader Reader) {
18 if (Reader.bytesRemaining() % sizeof(FrameData) != 0) {

24 return make_error<CodeViewError>(cv_error_code::corrupt_record,
