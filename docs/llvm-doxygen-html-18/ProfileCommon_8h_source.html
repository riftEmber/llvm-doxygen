105/// This is called when a count is seen in the profile.
106void ProfileSummaryBuilder::addCount(uint64_t Count) {
107 TotalCount += Count;
108 if (Count > MaxCount)
109 MaxCount = Count;
110 NumCounts++;
111 CountFrequencies[Count]++;
112}
