35bool AMDGPUInstrInfo::isUniformMMO(const MachineMemOperand *MMO) {
36 const Value *Ptr = MMO->getValue();
37 // UndefValue means this is a load of a kernel input. These are uniform.
38 // Sometimes LDS instructions have constant pointers.
39 // If Ptr is null, then that means this mem operand contains a
40 // PseudoSourceValue like GOT.
41 if (!Ptr || isa<UndefValue>(Ptr) ||
42 isa<Constant>(Ptr) || isa<GlobalValue>(Ptr))
43 return true;
44
45 if (MMO->getAddrSpace() == AMDGPUAS::CONSTANT_ADDRESS_32BIT)
46 return true;
47
48 if (const Argument *Arg = dyn_cast<Argument>(Ptr))
49 return AMDGPU::isArgPassedInSGPR(Arg);
50
51 const Instruction *I = dyn_cast<Instruction>(Ptr);
52 return I && I->getMetadata("amdgpu.uniform");
53}
