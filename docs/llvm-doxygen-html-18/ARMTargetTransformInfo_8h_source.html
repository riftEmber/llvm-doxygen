339/// isVREVMask - Check if a vector shuffle corresponds to a VREV
340/// instruction with the specified blocksize. (The order of the elements
341/// within each block of the vector is reversed.)
342inline bool isVREVMask(ArrayRef<int> M, EVT VT, unsigned BlockSize) {
343 assert((BlockSize == 16 || BlockSize == 32 || BlockSize == 64) &&
344 "Only possible block sizes for VREV are: 16, 32, 64");
345
346 unsigned EltSz = VT.getScalarSizeInBits();
347 if (EltSz != 8 && EltSz != 16 && EltSz != 32)
348 return false;
349
350 unsigned BlockElts = M[0] + 1;
351 // If the first shuffle index is UNDEF, be optimistic.
352 if (M[0] < 0)
353 BlockElts = BlockSize / EltSz;
354
355 if (BlockSize <= EltSz || BlockSize != BlockElts * EltSz)
356 return false;
357
358 for (unsigned i = 0, e = M.size(); i < e; ++i) {
359 if (M[i] < 0)
360 continue; // ignore UNDEF indices
361 if ((unsigned)M[i] != (i - i % BlockElts) + (BlockElts - 1 - i % BlockElts))
362 return false;
363 }
364
365 return true;
366}
