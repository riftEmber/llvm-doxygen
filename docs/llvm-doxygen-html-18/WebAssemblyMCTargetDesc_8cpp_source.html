118// Force static initialization.
119extern "C" LLVM_EXTERNAL_VISIBILITY void LLVMInitializeWebAssemblyTargetMC() {
120 for (Target *T :
121 {&getTheWebAssemblyTarget32(), &getTheWebAssemblyTarget64()}) {
122 // Register the MC asm info.
123 RegisterMCAsmInfoFn X(*T, createMCAsmInfo);
124
125 // Register the MC instruction info.
126 TargetRegistry::RegisterMCInstrInfo(*T, createMCInstrInfo);
127
128 // Register the MC register info.
129 TargetRegistry::RegisterMCRegInfo(*T, createMCRegisterInfo);
130
131 // Register the MCInstPrinter.
132 TargetRegistry::RegisterMCInstPrinter(*T, createMCInstPrinter);
133
134 // Register the MC code emitter.
135 TargetRegistry::RegisterMCCodeEmitter(*T, createCodeEmitter);
136
137 // Register the ASM Backend.
138 TargetRegistry::RegisterMCAsmBackend(*T, createAsmBackend);
139
140 // Register the MC subtarget info.
141 TargetRegistry::RegisterMCSubtargetInfo(*T, createMCSubtargetInfo);
142
143 // Register the object target streamer.
144 TargetRegistry::RegisterObjectTargetStreamer(*T,
145 createObjectTargetStreamer);
146 // Register the asm target streamer.
147 TargetRegistry::RegisterAsmTargetStreamer(*T, createAsmTargetStreamer);
148 // Register the null target streamer.
149 TargetRegistry::RegisterNullTargetStreamer(*T, createNullTargetStreamer);
150 }
151}
