21bool convertYAML(yaml::Input &YIn, raw_ostream &Out, ErrorHandler ErrHandler,
22 unsigned DocNum, uint64_t MaxSize) {
23 unsigned CurDocNum = 0;
24 do {
25 if (++CurDocNum != DocNum)
26 continue;
27
28 yaml::YamlObjectFile Doc;
29 YIn >> Doc;
30 if (std::error_code EC = YIn.error()) {
31 ErrHandler("failed to parse YAML input: " + EC.message());
32 return false;
33 }
34
35 if (Doc.Arch)
36 return yaml2archive(*Doc.Arch, Out, ErrHandler);
37 if (Doc.Elf)
38 return yaml2elf(*Doc.Elf, Out, ErrHandler, MaxSize);
39 if (Doc.Coff)
40 return yaml2coff(*Doc.Coff, Out, ErrHandler);
41 if (Doc.Goff)
42 return yaml2goff(*Doc.Goff, Out, ErrHandler);
43 if (Doc.MachO || Doc.FatMachO)
44 return yaml2macho(Doc, Out, ErrHandler);
45 if (Doc.Minidump)
46 return yaml2minidump(*Doc.Minidump, Out, ErrHandler);
47 if (Doc.Offload)
48 return yaml2offload(*Doc.Offload, Out, ErrHandler);
49 if (Doc.Wasm)
50 return yaml2wasm(*Doc.Wasm, Out, ErrHandler);
51 if (Doc.Xcoff)
52 return yaml2xcoff(*Doc.Xcoff, Out, ErrHandler);
53 if (Doc.DXContainer)
54 return yaml2dxcontainer(*Doc.DXContainer, Out, ErrHandler);
55
56 ErrHandler("unknown document type");
57 return false;
58
59 } while (YIn.nextDocument());
60
61 ErrHandler("cannot find the " + Twine(DocNum) +
62 getOrdinalSuffix(DocNum).data() + " document");
63 return false;
64}

66std::unique_ptr<object::ObjectFile>
67yaml2ObjectFile(SmallVectorImpl<char> &Storage, StringRef Yaml,
68 ErrorHandler ErrHandler) {
69 Storage.clear();
70 raw_svector_ostream OS(Storage);
71
72 yaml::Input YIn(Yaml);
73 if (!convertYAML(YIn, OS, ErrHandler))
74 return {};
75
76 Expected<std::unique_ptr<object::ObjectFile>> ObjOrErr =
77 object::ObjectFile::createObjectFile(
78 MemoryBufferRef(OS.str(), "YamlObject"));
79 if (ObjOrErr)
80 return std::move(*ObjOrErr);
81
82 ErrHandler(toString(ObjOrErr.takeError()));
83 return {};
84}
