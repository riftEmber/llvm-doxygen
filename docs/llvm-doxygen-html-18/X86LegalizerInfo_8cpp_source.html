103 switch (Q.Types[BigTyIdx].getSizeInBits()) {
104 case 16:
105 case 32:
106 case 64:
107 case 128:
108 case 256:
109 case 512:
110 break;
111 default:
112 return false;
113 }

115 case 8:
116 case 16:
117 case 32:
118 case 64:
119 case 128:
120 case 256:
121 return true;
122 default:
123 return false;
124 }
