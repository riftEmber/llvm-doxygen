90void AArch64Arm64ECCallLowering::getThunkType(FunctionType *FT,
91 AttributeList AttrList,
92 bool EntryThunk, raw_ostream &Out,
93 FunctionType *&Arm64Ty,
94 FunctionType *&X64Ty) {
95 Out << (EntryThunk ? "$ientry_thunk$cdecl$" : "$iexit_thunk$cdecl$");
96
97 Type *Arm64RetTy;
98 Type *X64RetTy;
99
100 SmallVector<Type *> Arm64ArgTypes;
101 SmallVector<Type *> X64ArgTypes;
102
103 // The first argument to a thunk is the called function, stored in x9.
104 // For exit thunks, we pass the called function down to the emulator;
105 // for entry thunks, we just call the Arm64 function directly.
106 if (!EntryThunk)
107 Arm64ArgTypes.push_back(PtrTy);
108 X64ArgTypes.push_back(PtrTy);
109
110 bool HasSretPtr = false;
111 getThunkRetType(FT, AttrList, Out, Arm64RetTy, X64RetTy, Arm64ArgTypes,
112 X64ArgTypes, HasSretPtr);
113
114 getThunkArgTypes(FT, AttrList, Out, Arm64ArgTypes, X64ArgTypes, HasSretPtr);
115
116 Arm64Ty = FunctionType::get(Arm64RetTy, Arm64ArgTypes, false);
117 X64Ty = FunctionType::get(X64RetTy, X64ArgTypes, false);
118}

120void AArch64Arm64ECCallLowering::getThunkArgTypes(
121 FunctionType *FT, AttributeList AttrList, raw_ostream &Out,
122 SmallVectorImpl<Type *> &Arm64ArgTypes,
123 SmallVectorImpl<Type *> &X64ArgTypes, bool HasSretPtr) {
124
125 Out << "$";
126 if (FT->isVarArg()) {
127 // We treat the variadic function's thunk as a normal function
128 // with the following type on the ARM side:
129 // rettype exitthunk(
130 // ptr x9, ptr x0, i64 x1, i64 x2, i64 x3, ptr x4, i64 x5)
131 //
132 // that can coverage all types of variadic function.
133 // x9 is similar to normal exit thunk, store the called function.
134 // x0-x3 is the arguments be stored in registers.
135 // x4 is the address of the arguments on the stack.
136 // x5 is the size of the arguments on the stack.
137 //
138 // On the x64 side, it's the same except that x5 isn't set.
139 //
140 // If both the ARM and X64 sides are sret, there are only three
141 // arguments in registers.
142 //
143 // If the X64 side is sret, but the ARM side isn't, we pass an extra value
144 // to/from the X64 side, and let SelectionDAG transform it into a memory
145 // location.
146 Out << "varargs";
147
148 // x0-x3
149 for (int i = HasSretPtr ? 1 : 0; i < 4; i++) {
150 Arm64ArgTypes.push_back(I64Ty);
151 X64ArgTypes.push_back(I64Ty);
152 }
153
154 // x4
155 Arm64ArgTypes.push_back(PtrTy);
156 X64ArgTypes.push_back(PtrTy);
157 // x5
158 Arm64ArgTypes.push_back(I64Ty);
159 // FIXME: x5 isn't actually passed/used by the x64 side; revisit once we
160 // have proper isel for varargs
161 X64ArgTypes.push_back(I64Ty);
162 return;
163 }
164
165 unsigned I = 0;
166 if (HasSretPtr)
167 I++;
168
169 if (I == FT->getNumParams()) {
170 Out << "v";
171 return;
172 }
173
174 for (unsigned E = FT->getNumParams(); I != E; ++I) {
175 Align ParamAlign = AttrList.getParamAlignment(I).valueOrOne();
176#if 0
177 // FIXME: Need more information about argument size; see
178 // https://reviews.llvm.org/D132926
179 uint64_t ArgSizeBytes = AttrList.getParamArm64ECArgSizeBytes(I);
180#else
181 uint64_t ArgSizeBytes = 0;
182#endif
183 Type *Arm64Ty, *X64Ty;
184 canonicalizeThunkType(FT->getParamType(I), ParamAlign,
185 /*Ret*/ false, ArgSizeBytes, Out, Arm64Ty, X64Ty);
186 Arm64ArgTypes.push_back(Arm64Ty);
187 X64ArgTypes.push_back(X64Ty);
188 }
189}

191void AArch64Arm64ECCallLowering::getThunkRetType(
192 FunctionType *FT, AttributeList AttrList, raw_ostream &Out,
193 Type *&Arm64RetTy, Type *&X64RetTy, SmallVectorImpl<Type *> &Arm64ArgTypes,
194 SmallVectorImpl<Type *> &X64ArgTypes, bool &HasSretPtr) {
195 Type *T = FT->getReturnType();
196#if 0
197 // FIXME: Need more information about argument size; see
198 // https://reviews.llvm.org/D132926
199 uint64_t ArgSizeBytes = AttrList.getRetArm64ECArgSizeBytes();
200#else
201 int64_t ArgSizeBytes = 0;
202#endif
203 if (T->isVoidTy()) {
204 if (FT->getNumParams()) {
205 auto SRetAttr = AttrList.getParamAttr(0, Attribute::StructRet);
206 auto InRegAttr = AttrList.getParamAttr(0, Attribute::InReg);
207 if (SRetAttr.isValid() && InRegAttr.isValid()) {
208 // sret+inreg indicates a call that returns a C++ class value. This is
209 // actually equivalent to just passing and returning a void* pointer
210 // as the first argument. Translate it that way, instead of trying
211 // to model "inreg" in the thunk's calling convention, to simplify
212 // the rest of the code.
213 Out << "i8";
214 Arm64RetTy = I64Ty;
215 X64RetTy = I64Ty;
216 return;
217 }
218 if (SRetAttr.isValid()) {
219 // FIXME: Sanity-check the sret type; if it's an integer or pointer,
220 // we'll get screwy mangling/codegen.
221 // FIXME: For large struct types, mangle as an integer argument and
222 // integer return, so we can reuse more thunks, instead of "m" syntax.
223 // (MSVC mangles this case as an integer return with no argument, but
224 // that's a miscompile.)
225 Type *SRetType = SRetAttr.getValueAsType();
226 Align SRetAlign = AttrList.getParamAlignment(0).valueOrOne();
227 Type *Arm64Ty, *X64Ty;
228 canonicalizeThunkType(SRetType, SRetAlign, /*Ret*/ true, ArgSizeBytes,
229 Out, Arm64Ty, X64Ty);
230 Arm64RetTy = VoidTy;
231 X64RetTy = VoidTy;
232 Arm64ArgTypes.push_back(FT->getParamType(0));
233 X64ArgTypes.push_back(FT->getParamType(0));
234 HasSretPtr = true;
235 return;
236 }
237 }
238
239 Out << "v";
240 Arm64RetTy = VoidTy;
241 X64RetTy = VoidTy;
242 return;
243 }
244
245 canonicalizeThunkType(T, Align(), /*Ret*/ true, ArgSizeBytes, Out, Arm64RetTy,
246 X64RetTy);
247 if (X64RetTy->isPointerTy()) {
248 // If the X64 type is canonicalized to a pointer, that means it's
249 // passed/returned indirectly. For a return value, that means it's an
250 // sret pointer.
251 X64ArgTypes.push_back(X64RetTy);
252 X64RetTy = VoidTy;
253 }

256void AArch64Arm64ECCallLowering::canonicalizeThunkType(
257 Type *T, Align Alignment, bool Ret, uint64_t ArgSizeBytes, raw_ostream &Out,
258 Type *&Arm64Ty, Type *&X64Ty) {
259 if (T->isFloatTy()) {
260 Out << "f";
261 Arm64Ty = T;
262 X64Ty = T;
263 return;
264 }
265
266 if (T->isDoubleTy()) {
267 Out << "d";
268 Arm64Ty = T;
269 X64Ty = T;
270 return;
271 }
272
273 if (T->isFloatingPointTy()) {
274 report_fatal_error(
275 "Only 32 and 64 bit floating points are supported for ARM64EC thunks");
276 }
277
278 auto &DL = M->getDataLayout();
279
280 if (auto *StructTy = dyn_cast<StructType>(T))
281 if (StructTy->getNumElements() == 1)
282 T = StructTy->getElementType(0);
283
284 if (T->isArrayTy()) {
285 Type *ElementTy = T->getArrayElementType();
286 uint64_t ElementCnt = T->getArrayNumElements();
287 uint64_t ElementSizePerBytes = DL.getTypeSizeInBits(ElementTy) / 8;
288 uint64_t TotalSizeBytes = ElementCnt * ElementSizePerBytes;
289 if (ElementTy->isFloatTy() || ElementTy->isDoubleTy()) {
290 Out << (ElementTy->isFloatTy() ? "F" : "D") << TotalSizeBytes;
291 if (Alignment.value() >= 8 && !T->isPointerTy())
292 Out << "a" << Alignment.value();
293 Arm64Ty = T;
294 if (TotalSizeBytes <= 8) {
295 // Arm64 returns small structs of float/double in float registers;
296 // X64 uses RAX.
297 X64Ty = llvm::Type::getIntNTy(M->getContext(), TotalSizeBytes * 8);
298 } else {
299 // Struct is passed directly on Arm64, but indirectly on X64.
300 X64Ty = PtrTy;
301 }
302 return;
303 } else if (T->isFloatingPointTy()) {
304 report_fatal_error("Only 32 and 64 bit floating points are supported for "
305 "ARM64EC thunks");
306 }
307 }
308
309 if ((T->isIntegerTy() || T->isPointerTy()) && DL.getTypeSizeInBits(T) <= 64) {
310 Out << "i8";
311 Arm64Ty = I64Ty;
312 X64Ty = I64Ty;
313 return;
314 }
315
316 unsigned TypeSize = ArgSizeBytes;
317 if (TypeSize == 0)
318 TypeSize = DL.getTypeSizeInBits(T) / 8;
319 Out << "m";
320 if (TypeSize != 4)
321 Out << TypeSize;
322 if (Alignment.value() >= 8 && !T->isPointerTy())
323 Out << "a" << Alignment.value();
324 // FIXME: Try to canonicalize Arm64Ty more thoroughly?
325 Arm64Ty = T;
326 if (TypeSize == 1 || TypeSize == 2 || TypeSize == 4 || TypeSize == 8) {
327 // Pass directly in an integer register
328 X64Ty = llvm::Type::getIntNTy(M->getContext(), TypeSize * 8);
329 } else {
330 // Passed directly on Arm64, but indirectly on X64.
331 X64Ty = PtrTy;
332 }
333}

335// This function builds the "exit thunk", a function which translates
336// arguments and return values when calling x64 code from AArch64 code.
337Function *AArch64Arm64ECCallLowering::buildExitThunk(FunctionType *FT,
338 AttributeList Attrs) {
339 SmallString<256> ExitThunkName;
340 llvm::raw_svector_ostream ExitThunkStream(ExitThunkName);
341 FunctionType *Arm64Ty, *X64Ty;
342 getThunkType(FT, Attrs, /*EntryThunk*/ false, ExitThunkStream, Arm64Ty,
343 X64Ty);
344 if (Function *F = M->getFunction(ExitThunkName))
345 return F;
346
347 Function *F = Function::Create(Arm64Ty, GlobalValue::LinkOnceODRLinkage, 0,
348 ExitThunkName, M);
349 F->setCallingConv(CallingConv::ARM64EC_Thunk_Native);
350 F->setSection(".wowthk$aa");
351 F->setComdat(M->getOrInsertComdat(ExitThunkName));
352 // Copy MSVC, and always set up a frame pointer. (Maybe this isn't necessary.)
353 F->addFnAttr("frame-pointer", "all");
354 // Only copy sret from the first argument. For C++ instance methods, clang can
355 // stick an sret marking on a later argument, but it doesn't actually affect
356 // the ABI, so we can omit it. This avoids triggering a verifier assertion.
357 if (FT->getNumParams()) {
358 auto SRet = Attrs.getParamAttr(0, Attribute::StructRet);
359 auto InReg = Attrs.getParamAttr(0, Attribute::InReg);
360 if (SRet.isValid() && !InReg.isValid())
361 F->addParamAttr(1, SRet);
362 }
363 // FIXME: Copy anything other than sret? Shouldn't be necessary for normal
364 // C ABI, but might show up in other cases.
365 BasicBlock *BB = BasicBlock::Create(M->getContext(), "", F);
366 IRBuilder<> IRB(BB);
367 Value *CalleePtr =
368 M->getOrInsertGlobal("__os_arm64x_dispatch_call_no_redirect", PtrTy);
369 Value *Callee = IRB.CreateLoad(PtrTy, CalleePtr);
370 auto &DL = M->getDataLayout();
371 SmallVector<Value *> Args;
372
373 // Pass the called function in x9.
374 Args.push_back(F->arg_begin());
375
376 Type *RetTy = Arm64Ty->getReturnType();
377 if (RetTy != X64Ty->getReturnType()) {
378 // If the return type is an array or struct, translate it. Values of size
379 // 8 or less go into RAX; bigger values go into memory, and we pass a
380 // pointer.
381 if (DL.getTypeStoreSize(RetTy) > 8) {
382 Args.push_back(IRB.CreateAlloca(RetTy));
383 }
384 }
385
386 for (auto &Arg : make_range(F->arg_begin() + 1, F->arg_end())) {
387 // Translate arguments from AArch64 calling convention to x86 calling
388 // convention.
389 //
390 // For simple types, we don't need to do any translation: they're
391 // represented the same way. (Implicit sign extension is not part of
392 // either convention.)
393 //
394 // The big thing we have to worry about is struct types... but
395 // fortunately AArch64 clang is pretty friendly here: the cases that need
396 // translation are always passed as a struct or array. (If we run into
397 // some cases where this doesn't work, we can teach clang to mark it up
398 // with an attribute.)
399 //
400 // The first argument is the called function, stored in x9.
401 if (Arg.getType()->isArrayTy() || Arg.getType()->isStructTy() ||
402 DL.getTypeStoreSize(Arg.getType()) > 8) {
403 Value *Mem = IRB.CreateAlloca(Arg.getType());
404 IRB.CreateStore(&Arg, Mem);
405 if (DL.getTypeStoreSize(Arg.getType()) <= 8) {
406 Type *IntTy = IRB.getIntNTy(DL.getTypeStoreSizeInBits(Arg.getType()));
407 Args.push_back(IRB.CreateLoad(IntTy, IRB.CreateBitCast(Mem, PtrTy)));
408 } else
409 Args.push_back(Mem);
410 } else {
411 Args.push_back(&Arg);
412 }
413 }
414 // FIXME: Transfer necessary attributes? sret? anything else?
415
416 Callee = IRB.CreateBitCast(Callee, PtrTy);
417 CallInst *Call = IRB.CreateCall(X64Ty, Callee, Args);
418 Call->setCallingConv(CallingConv::ARM64EC_Thunk_X64);
419
420 Value *RetVal = Call;
421 if (RetTy != X64Ty->getReturnType()) {
422 // If we rewrote the return type earlier, convert the return value to
423 // the proper type.
424 if (DL.getTypeStoreSize(RetTy) > 8) {
425 RetVal = IRB.CreateLoad(RetTy, Args[1]);
426 } else {
427 Value *CastAlloca = IRB.CreateAlloca(RetTy);
428 IRB.CreateStore(Call, IRB.CreateBitCast(CastAlloca, PtrTy));
429 RetVal = IRB.CreateLoad(RetTy, CastAlloca);
430 }
431 }
432
433 if (RetTy->isVoidTy())
434 IRB.CreateRetVoid();
435 else
436 IRB.CreateRet(RetVal);
437 return F;
438}

440// This function builds the "entry thunk", a function which translates
441// arguments and return values when calling AArch64 code from x64 code.
442Function *AArch64Arm64ECCallLowering::buildEntryThunk(Function *F) {
443 SmallString<256> EntryThunkName;
444 llvm::raw_svector_ostream EntryThunkStream(EntryThunkName);
445 FunctionType *Arm64Ty, *X64Ty;
446 getThunkType(F->getFunctionType(), F->getAttributes(), /*EntryThunk*/ true,
447 EntryThunkStream, Arm64Ty, X64Ty);
448 if (Function *F = M->getFunction(EntryThunkName))
449 return F;
450
451 Function *Thunk = Function::Create(X64Ty, GlobalValue::LinkOnceODRLinkage, 0,
452 EntryThunkName, M);
453 Thunk->setCallingConv(CallingConv::ARM64EC_Thunk_X64);
454 Thunk->setSection(".wowthk$aa");
455 Thunk->setComdat(M->getOrInsertComdat(EntryThunkName));
456 // Copy MSVC, and always set up a frame pointer. (Maybe this isn't necessary.)
457 Thunk->addFnAttr("frame-pointer", "all");
458
459 auto &DL = M->getDataLayout();
460 BasicBlock *BB = BasicBlock::Create(M->getContext(), "", Thunk);
461 IRBuilder<> IRB(BB);
462
463 Type *RetTy = Arm64Ty->getReturnType();
464 Type *X64RetType = X64Ty->getReturnType();
465
466 bool TransformDirectToSRet = X64RetType->isVoidTy() && !RetTy->isVoidTy();
467 unsigned ThunkArgOffset = TransformDirectToSRet ? 2 : 1;
468
469 // Translate arguments to call.
470 SmallVector<Value *> Args;
471 for (unsigned i = ThunkArgOffset, e = Thunk->arg_size(); i != e; ++i) {
472 Value *Arg = Thunk->getArg(i);
473 Type *ArgTy = Arm64Ty->getParamType(i - ThunkArgOffset);
474 if (ArgTy->isArrayTy() || ArgTy->isStructTy() ||
475 DL.getTypeStoreSize(ArgTy) > 8) {
476 // Translate array/struct arguments to the expected type.
477 if (DL.getTypeStoreSize(ArgTy) <= 8) {
478 Value *CastAlloca = IRB.CreateAlloca(ArgTy);
479 IRB.CreateStore(Arg, IRB.CreateBitCast(CastAlloca, PtrTy));
480 Arg = IRB.CreateLoad(ArgTy, CastAlloca);
481 } else {
482 Arg = IRB.CreateLoad(ArgTy, IRB.CreateBitCast(Arg, PtrTy));
483 }
484 }
485 Args.push_back(Arg);
486 }
487
488 // Call the function passed to the thunk.
489 Value *Callee = Thunk->getArg(0);
490 Callee = IRB.CreateBitCast(Callee, PtrTy);
491 Value *Call = IRB.CreateCall(Arm64Ty, Callee, Args);
492
493 Value *RetVal = Call;
494 if (TransformDirectToSRet) {
495 IRB.CreateStore(RetVal, IRB.CreateBitCast(Thunk->getArg(1), PtrTy));
496 } else if (X64RetType != RetTy) {
497 Value *CastAlloca = IRB.CreateAlloca(X64RetType);
498 IRB.CreateStore(Call, IRB.CreateBitCast(CastAlloca, PtrTy));
499 RetVal = IRB.CreateLoad(X64RetType, CastAlloca);
500 }
501
502 // Return to the caller. Note that the isel has code to translate this
503 // "ret" to a tail call to __os_arm64x_dispatch_ret. (Alternatively, we
504 // could emit a tail call here, but that would require a dedicated calling
505 // convention, which seems more complicated overall.)
506 if (X64RetType->isVoidTy())
507 IRB.CreateRetVoid();
508 else
509 IRB.CreateRet(RetVal);
510
511 return Thunk;
512}

514// Builds the "guest exit thunk", a helper to call a function which may or may
515// not be an exit thunk. (We optimistically assume non-dllimport function
516// declarations refer to functions defined in AArch64 code; if the linker
517// can't prove that, we use this routine instead.)
518Function *AArch64Arm64ECCallLowering::buildGuestExitThunk(Function *F) {
519 llvm::raw_null_ostream NullThunkName;
520 FunctionType *Arm64Ty, *X64Ty;
521 getThunkType(F->getFunctionType(), F->getAttributes(), /*EntryThunk*/ true,
522 NullThunkName, Arm64Ty, X64Ty);
523 auto MangledName = getArm64ECMangledFunctionName(F->getName().str());
524 assert(MangledName && "Can't guest exit to function that's already native");
525 std::string ThunkName = *MangledName;
526 if (ThunkName[0] == '?' && ThunkName.find("@") != std::string::npos) {
527 ThunkName.insert(ThunkName.find("@"), "$exit_thunk");
528 } else {
529 ThunkName.append("$exit_thunk");
530 }
531 Function *GuestExit =
532 Function::Create(Arm64Ty, GlobalValue::WeakODRLinkage, 0, ThunkName, M);
533 GuestExit->setComdat(M->getOrInsertComdat(ThunkName));
534 GuestExit->setSection(".wowthk$aa");
535 GuestExit->setMetadata(
536 "arm64ec_unmangled_name",
537 MDNode::get(M->getContext(),
538 MDString::get(M->getContext(), F->getName())));
539 GuestExit->setMetadata(
540 "arm64ec_ecmangled_name",
541 MDNode::get(M->getContext(),
542 MDString::get(M->getContext(), *MangledName)));
543 F->setMetadata("arm64ec_hasguestexit", MDNode::get(M->getContext(), {}));
544 BasicBlock *BB = BasicBlock::Create(M->getContext(), "", GuestExit);
545 IRBuilder<> B(BB);
546
547 // Load the global symbol as a pointer to the check function.
548 Value *GuardFn;
549 if (cfguard_module_flag == 2 && !F->hasFnAttribute("guard_nocf"))
550 GuardFn = GuardFnCFGlobal;
551 else
552 GuardFn = GuardFnGlobal;
553 LoadInst *GuardCheckLoad = B.CreateLoad(GuardFnPtrType, GuardFn);
554
555 // Create new call instruction. The CFGuard check should always be a call,
556 // even if the original CallBase is an Invoke or CallBr instruction.
557 Function *Thunk = buildExitThunk(F->getFunctionType(), F->getAttributes());
558 CallInst *GuardCheck = B.CreateCall(
559 GuardFnType, GuardCheckLoad,
560 {B.CreateBitCast(F, B.getPtrTy()), B.CreateBitCast(Thunk, B.getPtrTy())});
561
562 // Ensure that the first argument is passed in the correct register.
563 GuardCheck->setCallingConv(CallingConv::CFGuard_Check);
564
565 Value *GuardRetVal = B.CreateBitCast(GuardCheck, PtrTy);
566 SmallVector<Value *> Args;
567 for (Argument &Arg : GuestExit->args())
568 Args.push_back(&Arg);
569 CallInst *Call = B.CreateCall(Arm64Ty, GuardRetVal, Args);
570 Call->setTailCallKind(llvm::CallInst::TCK_MustTail);
571
572 if (Call->getType()->isVoidTy())
573 B.CreateRetVoid();
574 else
575 B.CreateRet(Call);
576
577 auto SRetAttr = F->getAttributes().getParamAttr(0, Attribute::StructRet);
578 auto InRegAttr = F->getAttributes().getParamAttr(0, Attribute::InReg);
579 if (SRetAttr.isValid() && !InRegAttr.isValid()) {
580 GuestExit->addParamAttr(0, SRetAttr);
581 Call->addParamAttr(0, SRetAttr);
582 }
583
584 return GuestExit;
585}

587// Lower an indirect call with inline code.
588void AArch64Arm64ECCallLowering::lowerCall(CallBase *CB) {
589 assert(Triple(CB->getModule()->getTargetTriple()).isOSWindows() &&
590 "Only applicable for Windows targets");
591
592 IRBuilder<> B(CB);
593 Value *CalledOperand = CB->getCalledOperand();
594
595 // If the indirect call is called within catchpad or cleanuppad,
596 // we need to copy "funclet" bundle of the call.
597 SmallVector<llvm::OperandBundleDef, 1> Bundles;
598 if (auto Bundle = CB->getOperandBundle(LLVMContext::OB_funclet))
599 Bundles.push_back(OperandBundleDef(*Bundle));
600
601 // Load the global symbol as a pointer to the check function.
602 Value *GuardFn;
603 if (cfguard_module_flag == 2 && !CB->hasFnAttr("guard_nocf"))
604 GuardFn = GuardFnCFGlobal;
605 else
606 GuardFn = GuardFnGlobal;
607 LoadInst *GuardCheckLoad = B.CreateLoad(GuardFnPtrType, GuardFn);
608
609 // Create new call instruction. The CFGuard check should always be a call,
610 // even if the original CallBase is an Invoke or CallBr instruction.
611 Function *Thunk = buildExitThunk(CB->getFunctionType(), CB->getAttributes());
612 CallInst *GuardCheck =
613 B.CreateCall(GuardFnType, GuardCheckLoad,
614 {B.CreateBitCast(CalledOperand, B.getPtrTy()),
615 B.CreateBitCast(Thunk, B.getPtrTy())},
616 Bundles);
617
618 // Ensure that the first argument is passed in the correct register.
619 GuardCheck->setCallingConv(CallingConv::CFGuard_Check);
620
621 Value *GuardRetVal = B.CreateBitCast(GuardCheck, CalledOperand->getType());
622 CB->setCalledOperand(GuardRetVal);
623}

625bool AArch64Arm64ECCallLowering::runOnModule(Module &Mod) {
626 if (!GenerateThunks)
627 return false;
628
629 M = &Mod;
630
631 // Check if this module has the cfguard flag and read its value.
632 if (auto *MD =
633 mdconst::extract_or_null<ConstantInt>(M->getModuleFlag("cfguard")))
634 cfguard_module_flag = MD->getZExtValue();
635
636 PtrTy = PointerType::getUnqual(M->getContext());
637 I64Ty = Type::getInt64Ty(M->getContext());
638 VoidTy = Type::getVoidTy(M->getContext());
639
640 GuardFnType = FunctionType::get(PtrTy, {PtrTy, PtrTy}, false);
641 GuardFnPtrType = PointerType::get(GuardFnType, 0);
642 GuardFnCFGlobal =
643 M->getOrInsertGlobal("__os_arm64x_check_icall_cfg", GuardFnPtrType);
644 GuardFnGlobal =
645 M->getOrInsertGlobal("__os_arm64x_check_icall", GuardFnPtrType);
646
647 SetVector<Function *> DirectCalledFns;
648 for (Function &F : Mod)
649 if (!F.isDeclaration() &&
650 F.getCallingConv() != CallingConv::ARM64EC_Thunk_Native &&
651 F.getCallingConv() != CallingConv::ARM64EC_Thunk_X64)
652 processFunction(F, DirectCalledFns);
653
654 struct ThunkInfo {
655 Constant *Src;
656 Constant *Dst;
657 unsigned Kind;
658 };
659 SmallVector<ThunkInfo> ThunkMapping;
660 for (Function &F : Mod) {
661 if (!F.isDeclaration() && (!F.hasLocalLinkage() || F.hasAddressTaken()) &&
662 F.getCallingConv() != CallingConv::ARM64EC_Thunk_Native &&
663 F.getCallingConv() != CallingConv::ARM64EC_Thunk_X64) {
664 if (!F.hasComdat())
665 F.setComdat(Mod.getOrInsertComdat(F.getName()));
666 ThunkMapping.push_back({&F, buildEntryThunk(&F), 1});
667 }
668 }
669 for (Function *F : DirectCalledFns) {
670 ThunkMapping.push_back(
671 {F, buildExitThunk(F->getFunctionType(), F->getAttributes()), 4});
672 if (!F->hasDLLImportStorageClass())
673 ThunkMapping.push_back({buildGuestExitThunk(F), F, 0});
674 }
675
676 if (!ThunkMapping.empty()) {
677 SmallVector<Constant *> ThunkMappingArrayElems;
678 for (ThunkInfo &Thunk : ThunkMapping) {
679 ThunkMappingArrayElems.push_back(ConstantStruct::getAnon(
680 {ConstantExpr::getBitCast(Thunk.Src, PtrTy),
681 ConstantExpr::getBitCast(Thunk.Dst, PtrTy),
682 ConstantInt::get(M->getContext(), APInt(32, Thunk.Kind))}));
683 }
684 Constant *ThunkMappingArray = ConstantArray::get(
685 llvm::ArrayType::get(ThunkMappingArrayElems[0]->getType(),
686 ThunkMappingArrayElems.size()),
687 ThunkMappingArrayElems);
688 new GlobalVariable(Mod, ThunkMappingArray->getType(), /*isConstant*/ false,
689 GlobalValue::ExternalLinkage, ThunkMappingArray,
690 "llvm.arm64ec.symbolmap");
691 }
692
693 return true;
694}

696bool AArch64Arm64ECCallLowering::processFunction(
697 Function &F, SetVector<Function *> &DirectCalledFns) {
698 SmallVector<CallBase *, 8> IndirectCalls;
699
700 // For ARM64EC targets, a function definition's name is mangled differently
701 // from the normal symbol. We currently have no representation of this sort
702 // of symbol in IR, so we change the name to the mangled name, then store
703 // the unmangled name as metadata. Later passes that need the unmangled
704 // name (emitting the definition) can grab it from the metadata.
705 //
706 // FIXME: Handle functions with weak linkage?
707 if (F.hasExternalLinkage() || F.hasWeakLinkage() || F.hasLinkOnceLinkage()) {
708 if (std::optional<std::string> MangledName =
709 getArm64ECMangledFunctionName(F.getName().str())) {
710 F.setMetadata("arm64ec_unmangled_name",
711 MDNode::get(M->getContext(),
712 MDString::get(M->getContext(), F.getName())));
713 if (F.hasComdat() && F.getComdat()->getName() == F.getName()) {
714 Comdat *MangledComdat = M->getOrInsertComdat(MangledName.value());
715 SmallVector<GlobalObject *> ComdatUsers =
716 to_vector(F.getComdat()->getUsers());
717 for (GlobalObject *User : ComdatUsers)
718 User->setComdat(MangledComdat);
719 }
720 F.setName(MangledName.value());
721 }
722 }
723
724 // Iterate over the instructions to find all indirect call/invoke/callbr
725 // instructions. Make a separate list of pointers to indirect
726 // call/invoke/callbr instructions because the original instructions will be
727 // deleted as the checks are added.
728 for (BasicBlock &BB : F) {
729 for (Instruction &I : BB) {
730 auto *CB = dyn_cast<CallBase>(&I);
731 if (!CB || CB->getCallingConv() == CallingConv::ARM64EC_Thunk_X64 ||
732 CB->isInlineAsm())
733 continue;
734
735 // We need to instrument any call that isn't directly calling an
736 // ARM64 function.
737 //
738 // FIXME: getCalledFunction() fails if there's a bitcast (e.g.
739 // unprototyped functions in C)
740 if (Function *F = CB->getCalledFunction()) {
741 if (!LowerDirectToIndirect || F->hasLocalLinkage() ||
742 F->isIntrinsic() || !F->isDeclaration())
743 continue;
744
745 DirectCalledFns.insert(F);
746 continue;
747 }
748
749 IndirectCalls.push_back(CB);
750 ++Arm64ECCallsLowered;
751 }
752 }
753
754 if (IndirectCalls.empty())
755 return false;
756
757 for (CallBase *CB : IndirectCalls)
758 lowerCall(CB);
759
760 return true;
761}
