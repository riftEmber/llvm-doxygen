71void FileWriter::alignTo(size_t Align) {
72 off_t Offset = OS.tell();
73 off_t AlignedOffset = (Offset + Align - 1) / Align * Align;
74 if (AlignedOffset == Offset)
75 return;
76 off_t PadCount = AlignedOffset - Offset;
