84bool RISCVInstructionSelector::select(MachineInstr &I) {
85
86 if (!isPreISelGenericOpcode(I.getOpcode())) {
87 // Certain non-generic instructions also need some special handling.
88 return true;
89 }
90
91 if (selectImpl(I, *CoverageInfo))
92 return true;
93
94 return false;
95}
