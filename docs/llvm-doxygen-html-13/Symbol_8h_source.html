27 None = 0,
28
29 /// Thread-local value symbol

32 /// Weak defined symbol

62using TargetList = SmallVector<Target, 5>;
63class Symbol {
64public:
65 Symbol(SymbolKind Kind, StringRef Name, TargetList Targets, SymbolFlags Flags)
66 : Name(Name), Targets(std::move(Targets)), Kind(Kind), Flags(Flags) {}
67
68 void addTarget(Target target) { Targets.emplace_back(target); }
69 SymbolKind getKind() const { return Kind; }
70 StringRef getName() const { return Name; }
71 ArchitectureSet getArchitectures() const {
72 return mapToArchitectureSet(Targets);
73 }
74 SymbolFlags getFlags() const { return Flags; }
75
76 bool isWeakDefined() const {
77 return (Flags & SymbolFlags::WeakDefined) == SymbolFlags::WeakDefined;
78 }
79
80 bool isWeakReferenced() const {
81 return (Flags & SymbolFlags::WeakReferenced) == SymbolFlags::WeakReferenced;
82 }
83
84 bool isThreadLocalValue() const {
85 return (Flags & SymbolFlags::ThreadLocalValue) ==
86 SymbolFlags::ThreadLocalValue;
87 }
88
89 bool isUndefined() const {
90 return (Flags & SymbolFlags::Undefined) == SymbolFlags::Undefined;
91 }
92
93 bool isReexported() const {
94 return (Flags & SymbolFlags::Rexported) == SymbolFlags::Rexported;
95 }

114 return std::tie(Name, Kind, Targets, Flags) ==
115 std::tie(O.Name, O.Kind, O.Targets, O.Flags);
116 }
117

119
