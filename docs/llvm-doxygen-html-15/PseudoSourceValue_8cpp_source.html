39bool PseudoSourceValue::isConstant(const MachineFrameInfo *) const {
40 if (isStack())
41 return false;
42 if (isGOT() || isConstantPool() || isJumpTable())
43 return true;
44 llvm_unreachable("Unknown PseudoSourceValue!");
45}
46
47bool PseudoSourceValue::isAliased(const MachineFrameInfo *) const {
48 if (isStack() || isGOT() || isConstantPool() || isJumpTable())
49 return false;
50 llvm_unreachable("Unknown PseudoSourceValue!");
51}
52
53bool PseudoSourceValue::mayAlias(const MachineFrameInfo *) const {
54 return !(isGOT() || isConstantPool() || isJumpTable());
55}

57bool FixedStackPseudoSourceValue::isConstant(
58 const MachineFrameInfo *MFI) const {
59 return MFI && MFI->isImmutableObjectIndex(FI);
60}
61
62bool FixedStackPseudoSourceValue::isAliased(const MachineFrameInfo *MFI) const {
63 if (!MFI)
64 return true;
65 return MFI->isAliasedObjectIndex(FI);
66}
67
68bool FixedStackPseudoSourceValue::mayAlias(const MachineFrameInfo *MFI) const {
69 if (!MFI)
70 return true;
71 // Spill slots will not alias any LLVM IR value.
72 return !MFI->isSpillSlotObjectIndex(FI);
73}
