236 // Set B[I] = C for every live bundle where B[I] was NoCand.
237 unsigned getBundles(SmallVectorImpl<unsigned> &B, unsigned C) {
238 unsigned Count = 0;
239 for (unsigned I : LiveBundles.set_bits())
240 if (B[I] == NoCand) {
241 B[I] = C;
242 Count++;
243 }
244 return Count;
245 }
