112void LLVMAddInternalizePass(LLVMPassManagerRef PM, unsigned AllButMain) {
113 auto PreserveMain = [=](const GlobalValue &GV) {
114 return AllButMain && GV.getName() == "main";
115 };
116 unwrap(PM)->add(createInternalizePass(PreserveMain));
117}

119void LLVMAddInternalizePassWithMustPreservePredicate(
120 LLVMPassManagerRef PM,
121 void *Context,
122 LLVMBool (*Pred)(LLVMValueRef, void *)) {
123 unwrap(PM)->add(createInternalizePass([=](const GlobalValue &GV) {
124 return Pred(wrap(&GV), Context) == 0 ? false : true;
125 }));
126}
