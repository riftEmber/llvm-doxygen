171// General helper for all the binary generic MI such as G_ADD/G_SUB etc
172template <typename LHS_P, typename RHS_P, unsigned Opcode,
173 bool Commutable = false>
174struct BinaryOp_match {
175 LHS_P L;
176 RHS_P R;
177
178 BinaryOp_match(const LHS_P &LHS, const RHS_P &RHS) : L(LHS), R(RHS) {}
179 template <typename OpTy>
180 bool match(const MachineRegisterInfo &MRI, OpTy &&Op) {
181 MachineInstr *TmpMI;
182 if (mi_match(Op, MRI, m_MInstr(TmpMI))) {
183 if (TmpMI->getOpcode() == Opcode && TmpMI->getNumOperands() == 3) {
184 return (L.match(MRI, TmpMI->getOperand(1).getReg()) &&
185 R.match(MRI, TmpMI->getOperand(2).getReg())) ||
186 (Commutable && (R.match(MRI, TmpMI->getOperand(1).getReg()) &&
187 L.match(MRI, TmpMI->getOperand(2).getReg())));
188 }
189 }
190 return false;
191 }
192};

336// General helper for generic MI compares, i.e. G_ICMP and G_FCMP
337// TODO: Allow checking a specific predicate.
338template <typename Pred_P, typename LHS_P, typename RHS_P, unsigned Opcode>
339struct CompareOp_match {
340 Pred_P P;
341 LHS_P L;
342 RHS_P R;
343
344 CompareOp_match(const Pred_P &Pred, const LHS_P &LHS, const RHS_P &RHS)
345 : P(Pred), L(LHS), R(RHS) {}
346
347 template <typename OpTy>
348 bool match(const MachineRegisterInfo &MRI, OpTy &&Op) {
349 MachineInstr *TmpMI;
350 if (!mi_match(Op, MRI, m_MInstr(TmpMI)) || TmpMI->getOpcode() != Opcode)
351 return false;
352
353 auto TmpPred =
354 static_cast<CmpInst::Predicate>(TmpMI->getOperand(1).getPredicate());
355 if (!P.match(MRI, TmpPred))
356 return false;
357
358 return L.match(MRI, TmpMI->getOperand(2).getReg()) &&
359 R.match(MRI, TmpMI->getOperand(3).getReg());
360 }
361};

365m_GICmp(const Pred &P, const LHS &L, const RHS &R) {
