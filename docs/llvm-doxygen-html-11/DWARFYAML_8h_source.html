28struct InitialLength {
29 uint32_t TotalLength;
30 uint64_t TotalLength64;
31
32 bool isDWARF64() const { return TotalLength == UINT32_MAX; }
33
34 uint64_t getLength() const {
35 return isDWARF64() ? TotalLength64 : TotalLength;
36 }
37
38 void setLength(uint64_t Len) {
39 if (Len >= (uint64_t)UINT32_MAX) {
40 TotalLength64 = Len;
41 TotalLength = UINT32_MAX;
42 } else {
43 TotalLength = Len;
44 }
45 }
46};
