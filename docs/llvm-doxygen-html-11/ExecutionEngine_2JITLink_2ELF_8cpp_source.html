30void jitLink_ELF(std::unique_ptr<JITLinkContext> Ctx) {
31
32 // We don't want to do full ELF validation here. We just verify it is elf'ish.
33 // Probably should parse into an elf header when we support more than x86 :)
34
35 StringRef Data = Ctx->getObjectBuffer().getBuffer();
36 if (Data.size() < llvm::ELF::EI_MAG3 + 1) {
37 Ctx->notifyFailed(make_error<JITLinkError>("Truncated ELF buffer"));
38 return;
39 }
40
41 if (!memcmp(Data.data(), llvm::ELF::ElfMagic, strlen(llvm::ELF::ElfMagic))) {
42 if (Data.data()[llvm::ELF::EI_CLASS] == ELF::ELFCLASS64) {
43 return jitLink_ELF_x86_64(std::move(Ctx));
44 }
45 }
46
47 Ctx->notifyFailed(make_error<JITLinkError>("ELF magic not valid"));
48}
