25static bool useCompactUnwind(const Triple &T) {
26 // Only on darwin.
27 if (!T.isOSDarwin())
28 return false;
29
30 // aarch64 always has it.
31 if (T.getArch() == Triple::aarch64 || T.getArch() == Triple::aarch64_32)
32 return true;
33
34 // armv7k always has it.
35 if (T.isWatchABI())
36 return true;
37
38 // Use it on newer version of OS X.
39 if (T.isMacOSX() && !T.isMacOSXVersionLT(10, 6))
40 return true;
41
42 // And the iOS simulator.
43 if (T.isiOS() && T.isX86())
44 return true;
45
46 return false;
47}

305 case Triple::mips:
306 case Triple::mipsel:
307 case Triple::mips64:

309 // We cannot use DW_EH_PE_sdata8 for the large PositionIndependent case
310 // since there is no R_MIPS_PC64 relocation (only a 32-bit version).
311 if (PositionIndependent && !Large)
312 FDECFIEncoding = dwarf::DW_EH_PE_pcrel | dwarf::DW_EH_PE_sdata4;

314 FDECFIEncoding = Ctx->getAsmInfo()->getCodePointerSize() == 4

500void MCObjectFileInfo::initCOFFMCObjectFileInfo(const Triple &T) {
501 EHFrameSection =
502 Ctx->getCOFFSection(".eh_frame", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
503 COFF::IMAGE_SCN_MEM_READ,
504 SectionKind::getData());
505
506 // Set the `IMAGE_SCN_MEM_16BIT` flag when compiling for thumb mode. This is
507 // used to indicate to the linker that the text segment contains thumb instructions
508 // and to set the ISA selection bit for calls accordingly.
509 const bool IsThumb = T.getArch() == Triple::thumb;
510
511 CommDirectiveSupportsAlignment = true;
512
513 // COFF
514 BSSSection = Ctx->getCOFFSection(
515 ".bss", COFF::IMAGE_SCN_CNT_UNINITIALIZED_DATA |
516 COFF::IMAGE_SCN_MEM_READ | COFF::IMAGE_SCN_MEM_WRITE,
517 SectionKind::getBSS());
518 TextSection = Ctx->getCOFFSection(
519 ".text",
520 (IsThumb ? COFF::IMAGE_SCN_MEM_16BIT : (COFF::SectionCharacteristics)0) |
521 COFF::IMAGE_SCN_CNT_CODE | COFF::IMAGE_SCN_MEM_EXECUTE |
522 COFF::IMAGE_SCN_MEM_READ,
523 SectionKind::getText());
524 DataSection = Ctx->getCOFFSection(
525 ".data", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ |
526 COFF::IMAGE_SCN_MEM_WRITE,
527 SectionKind::getData());
528 ReadOnlySection = Ctx->getCOFFSection(
529 ".rdata", COFF::IMAGE_SCN_CNT_INITIALIZED_DATA | COFF::IMAGE_SCN_MEM_READ,
530 SectionKind::getReadOnly());
531
532 if (T.getArch() == Triple::x86_64 || T.getArch() == Triple::aarch64) {
533 // On Windows 64 with SEH, the LSDA is emitted into the .xdata section
534 LSDASection = nullptr;
535 } else {
536 LSDASection = Ctx->getCOFFSection(".gcc_except_table",
537 COFF::IMAGE_SCN_CNT_INITIALIZED_DATA |
538 COFF::IMAGE_SCN_MEM_READ,
539 SectionKind::getReadOnly());
