8/// \file
9///
10/// This header provides classes for managing a pipeline of passes over loops
11/// in LLVM IR.
12///
13/// The primary loop pass pipeline is managed in a very particular way to
14/// provide a set of core guarantees:
15/// 1) Loops are, where possible, in simplified form.
16/// 2) Loops are *always* in LCSSA form.
17/// 3) A collection of Loop-specific analysis results are available:
18/// - LoopInfo
19/// - DominatorTree
20/// - ScalarEvolution
21/// - AAManager
22/// 4) All loop passes preserve #1 (where possible), #2, and #3.
23/// 5) Loop passes run over each loop in the loop nest from the innermost to
24/// the outermost. Specifically, all inner loops are processed before
25/// passes run over outer loops. When running the pipeline across an inner
26/// loop creates new inner loops, those are added and processed in this
27/// order as well.
28///
29/// This process is designed to facilitate transformations which simplify,
30/// reduce, and remove loops. For passes which are more oriented towards
31/// optimizing loops, especially optimizing loop *nests* instead of single
32/// loops in isolation, this framework is less interesting.
33///
34//===----------------------------------------------------------------------===//

108/// This class provides an interface for updating the loop pass manager based
109/// on mutations to the loop nest.
110///
111/// A reference to an instance of this class is passed as an argument to each
112/// Loop pass, and Loop passes should use it to update LPM infrastructure if
113/// they modify the loop nest structure.
114class LPMUpdater {
115public:
116 /// This can be queried by loop passes which run other loop passes (like pass
117 /// managers) to know whether the loop needs to be skipped due to updates to
118 /// the loop nest.
119 ///
120 /// If this returns true, the loop object may have been deleted, so passes
121 /// should take care not to touch the object.
122 bool skipCurrentLoop() const { return SkipCurrentLoop; }
123
124 /// Loop passes should use this method to indicate they have deleted a loop
125 /// from the nest.
126 ///
127 /// Note that this loop must either be the current loop or a subloop of the
128 /// current loop. This routine must be called prior to removing the loop from
129 /// the loop nest.
130 ///
131 /// If this is called for the current loop, in addition to clearing any
132 /// state, this routine will mark that the current loop should be skipped by
133 /// the rest of the pass management infrastructure.
134 void markLoopAsDeleted(Loop &L, llvm::StringRef Name) {
135 LAM.clear(L, Name);
136 assert((&L == CurrentL || CurrentL->contains(&L)) &&
137 "Cannot delete a loop outside of the "
138 "subloop tree currently being processed.");
139 if (&L == CurrentL)
140 SkipCurrentLoop = true;
141 }
142
143 /// Loop passes should use this method to indicate they have added new child
144 /// loops of the current loop.
145 ///
146 /// \p NewChildLoops must contain only the immediate children. Any nested
147 /// loops within them will be visited in postorder as usual for the loop pass
148 /// manager.
149 void addChildLoops(ArrayRef<Loop *> NewChildLoops) {
150 // Insert ourselves back into the worklist first, as this loop should be
151 // revisited after all the children have been processed.
152 Worklist.insert(CurrentL);
153
154#ifndef NDEBUG
155 for (Loop *NewL : NewChildLoops)
156 assert(NewL->getParentLoop() == CurrentL && "All of the new loops must "
157 "be immediate children of "
158 "the current loop!");
159#endif
160
161 appendLoopsToWorklist(NewChildLoops, Worklist);
162
163 // Also skip further processing of the current loop--it will be revisited
164 // after all of its newly added children are accounted for.
165 SkipCurrentLoop = true;
166 }
167
168 /// Loop passes should use this method to indicate they have added new
169 /// sibling loops to the current loop.
170 ///
171 /// \p NewSibLoops must only contain the immediate sibling loops. Any nested
172 /// loops within them will be visited in postorder as usual for the loop pass
173 /// manager.
174 void addSiblingLoops(ArrayRef<Loop *> NewSibLoops) {
175#ifndef NDEBUG
176 for (Loop *NewL : NewSibLoops)
177 assert(NewL->getParentLoop() == ParentL &&
178 "All of the new loops must be siblings of the current loop!");
179#endif
180
181 appendLoopsToWorklist(NewSibLoops, Worklist);
182
183 // No need to skip the current loop or revisit it, as sibling loops
184 // shouldn't impact anything.
185 }
186
187 /// Restart the current loop.
188 ///
189 /// Loop passes should call this method to indicate the current loop has been
190 /// sufficiently changed that it should be re-visited from the begining of
191 /// the loop pass pipeline rather than continuing.
192 void revisitCurrentLoop() {
193 // Tell the currently in-flight pipeline to stop running.
194 SkipCurrentLoop = true;
195
196 // And insert ourselves back into the worklist.
197 Worklist.insert(CurrentL);
198 }
199
200private:
201 template <typename LoopPassT> friend class llvm::FunctionToLoopPassAdaptor;
202
203 /// The \c FunctionToLoopPassAdaptor's worklist of loops to process.
204 SmallPriorityWorklist<Loop *, 4> &Worklist;
205
206 /// The analysis manager for use in the current loop nest.
207 LoopAnalysisManager &LAM;
208
209 Loop *CurrentL;
210 bool SkipCurrentLoop;
211
212#ifndef NDEBUG
213 // In debug builds we also track the parent loop to implement asserts even in
214 // the face of loop deletion.
215 Loop *ParentL;
216#endif
217
218 LPMUpdater(SmallPriorityWorklist<Loop *, 4> &Worklist,
219 LoopAnalysisManager &LAM)
220 : Worklist(Worklist), LAM(LAM) {}
221};

223/// Adaptor that maps from a function to its loops.
224///
225/// Designed to allow composition of a LoopPass(Manager) and a
226/// FunctionPassManager. Note that if this pass is constructed with a \c
227/// FunctionAnalysisManager it will run the \c LoopAnalysisManagerFunctionProxy
228/// analysis prior to running the loop passes over the function to enable a \c
229/// LoopAnalysisManager to be used within this run safely.
230template <typename LoopPassT>
231class FunctionToLoopPassAdaptor
232 : public PassInfoMixin<FunctionToLoopPassAdaptor<LoopPassT>> {
233public:
234 explicit FunctionToLoopPassAdaptor(LoopPassT Pass, bool UseMemorySSA = false,
235 bool DebugLogging = false)
236 : Pass(std::move(Pass)), LoopCanonicalizationFPM(DebugLogging),
237 UseMemorySSA(UseMemorySSA) {
238 LoopCanonicalizationFPM.addPass(LoopSimplifyPass());
239 LoopCanonicalizationFPM.addPass(LCSSAPass());
240 }
241
242 /// Runs the loop passes across every loop in the function.
243 PreservedAnalyses run(Function &F, FunctionAnalysisManager &AM) {
244 // Before we even compute any loop analyses, first run a miniature function
245 // pass pipeline to put loops into their canonical form. Note that we can
246 // directly build up function analyses after this as the function pass
247 // manager handles all the invalidation at that layer.
248 PassInstrumentation PI = AM.getResult<PassInstrumentationAnalysis>(F);
249
250 PreservedAnalyses PA = PreservedAnalyses::all();
251 // Check the PassInstrumentation's BeforePass callbacks before running the
252 // canonicalization pipeline.
253 if (PI.runBeforePass<Function>(LoopCanonicalizationFPM, F)) {
254 PA = LoopCanonicalizationFPM.run(F, AM);
255 PI.runAfterPass<Function>(LoopCanonicalizationFPM, F);
256 }
257
258 // Get the loop structure for this function
259 LoopInfo &LI = AM.getResult<LoopAnalysis>(F);
260
261 // If there are no loops, there is nothing to do here.
262 if (LI.empty())
263 return PA;
264
265 // Get the analysis results needed by loop passes.
266 MemorySSA *MSSA = UseMemorySSA
267 ? (&AM.getResult<MemorySSAAnalysis>(F).getMSSA())
268 : nullptr;
269 LoopStandardAnalysisResults LAR = {AM.getResult<AAManager>(F),
270 AM.getResult<AssumptionAnalysis>(F),
271 AM.getResult<DominatorTreeAnalysis>(F),
272 AM.getResult<LoopAnalysis>(F),
273 AM.getResult<ScalarEvolutionAnalysis>(F),
274 AM.getResult<TargetLibraryAnalysis>(F),
275 AM.getResult<TargetIRAnalysis>(F),
276 MSSA};
277
278 // Setup the loop analysis manager from its proxy. It is important that
279 // this is only done when there are loops to process and we have built the
280 // LoopStandardAnalysisResults object. The loop analyses cached in this
281 // manager have access to those analysis results and so it must invalidate
282 // itself when they go away.
283 auto &LAMFP = AM.getResult<LoopAnalysisManagerFunctionProxy>(F);
284 if (UseMemorySSA)
285 LAMFP.markMSSAUsed();
286 LoopAnalysisManager &LAM = LAMFP.getManager();
287
288 // A postorder worklist of loops to process.
289 SmallPriorityWorklist<Loop *, 4> Worklist;
290
291 // Register the worklist and loop analysis manager so that loop passes can
292 // update them when they mutate the loop nest structure.
293 LPMUpdater Updater(Worklist, LAM);
294
295 // Add the loop nests in the reverse order of LoopInfo. See method
296 // declaration.
297 appendLoopsToWorklist(LI, Worklist);
298
299 do {
300 Loop *L = Worklist.pop_back_val();
301
302 // Reset the update structure for this loop.
303 Updater.CurrentL = L;
304 Updater.SkipCurrentLoop = false;
305
306#ifndef NDEBUG
307 // Save a parent loop pointer for asserts.
308 Updater.ParentL = L->getParentLoop();
309
310 // Verify the loop structure and LCSSA form before visiting the loop.
311 L->verifyLoop();
312 assert(L->isRecursivelyLCSSAForm(LAR.DT, LI) &&
313 "Loops must remain in LCSSA form!");
314#endif
315 // Check the PassInstrumentation's BeforePass callbacks before running the
316 // pass, skip its execution completely if asked to (callback returns
317 // false).
318 if (!PI.runBeforePass<Loop>(Pass, *L))
319 continue;
320
321 PreservedAnalyses PassPA;
322 {
323 TimeTraceScope TimeScope(Pass.name());
324 PassPA = Pass.run(*L, LAM, LAR, Updater);
325 }
326
327 // Do not pass deleted Loop into the instrumentation.
328 if (Updater.skipCurrentLoop())
329 PI.runAfterPassInvalidated<Loop>(Pass);
330 else
331 PI.runAfterPass<Loop>(Pass, *L);
332
333 // FIXME: We should verify the set of analyses relevant to Loop passes
334 // are preserved.
335
336 // If the loop hasn't been deleted, we need to handle invalidation here.
337 if (!Updater.skipCurrentLoop())
338 // We know that the loop pass couldn't have invalidated any other
339 // loop's analyses (that's the contract of a loop pass), so directly
340 // handle the loop analysis manager's invalidation here.
341 LAM.invalidate(*L, PassPA);
342
343 // Then intersect the preserved set so that invalidation of module
344 // analyses will eventually occur when the module pass completes.
345 PA.intersect(std::move(PassPA));
346 } while (!Worklist.empty());
347
348 // By definition we preserve the proxy. We also preserve all analyses on
349 // Loops. This precludes *any* invalidation of loop analyses by the proxy,
350 // but that's OK because we've taken care to invalidate analyses in the
351 // loop analysis manager incrementally above.
352 PA.preserveSet<AllAnalysesOn<Loop>>();
353 PA.preserve<LoopAnalysisManagerFunctionProxy>();
354 // We also preserve the set of standard analyses.
355 PA.preserve<DominatorTreeAnalysis>();
356 PA.preserve<LoopAnalysis>();
357 PA.preserve<ScalarEvolutionAnalysis>();
358 if (UseMemorySSA)
359 PA.preserve<MemorySSAAnalysis>();
360 // FIXME: What we really want to do here is preserve an AA category, but
361 // that concept doesn't exist yet.
362 PA.preserve<AAManager>();
363 PA.preserve<BasicAA>();
364 PA.preserve<GlobalsAA>();
365 PA.preserve<SCEVAA>();
366 return PA;
367 }
368
369private:
370 LoopPassT Pass;
371
372 FunctionPassManager LoopCanonicalizationFPM;
373
374 bool UseMemorySSA = false;
375};
