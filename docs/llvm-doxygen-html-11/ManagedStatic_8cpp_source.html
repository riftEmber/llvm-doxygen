33void ManagedStaticBase::RegisterManagedStatic(void *(*Creator)(),
34 void (*Deleter)(void*)) const {
35 assert(Creator);
36 if (llvm_is_multithreaded()) {
37 std::lock_guard<std::recursive_mutex> Lock(*getManagedStaticMutex());
38
39 if (!Ptr.load(std::memory_order_relaxed)) {
40 void *Tmp = Creator();
41
42 Ptr.store(Tmp, std::memory_order_release);
43 DeleterFn = Deleter;
44
45 // Add to list of managed statics.
46 Next = StaticList;
47 StaticList = this;
48 }
49 } else {
50 assert(!Ptr && !DeleterFn && !Next &&
51 "Partially initialized ManagedStatic!?");
52 Ptr = Creator();
53 DeleterFn = Deleter;
54
55 // Add to list of managed statics.
56 Next = StaticList;
57 StaticList = this;
58 }
59}

61void ManagedStaticBase::destroy() const {
62 assert(DeleterFn && "ManagedStatic not initialized correctly!");
63 assert(StaticList == this &&
64 "Not destroyed in reverse order of construction?");
65 // Unlink from list.
66 StaticList = Next;
67 Next = nullptr;
68
69 // Destroy memory.
70 DeleterFn(Ptr);
71
72 // Cleanup.
73 Ptr = nullptr;
74 DeleterFn = nullptr;
75}

77/// llvm_shutdown - Deallocate and destroy all ManagedStatic variables.
78void llvm::llvm_shutdown() {

80
81 while (StaticList)
82 StaticList->destroy();
83}
