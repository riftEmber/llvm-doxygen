280template <typename JITType, typename SetterImpl, typename State>
281class LLJITBuilderSetters {
282public:
283
284 /// Set an ExecutionSession for this instance.
285 SetterImpl &setExecutionSession(std::unique_ptr<ExecutionSession> ES) {
286 impl().ES = std::move(ES);
287 return impl();
288 }
289
290 /// Set the JITTargetMachineBuilder for this instance.
291 ///
292 /// If this method is not called, JITTargetMachineBuilder::detectHost will be
293 /// used to construct a default target machine builder for the host platform.
294 SetterImpl &setJITTargetMachineBuilder(JITTargetMachineBuilder JTMB) {
295 impl().JTMB = std::move(JTMB);
296 return impl();
297 }
298
299 /// Return a reference to the JITTargetMachineBuilder.
300 ///
301 Optional<JITTargetMachineBuilder> &getJITTargetMachineBuilder() {
302 return impl().JTMB;
303 }
304
305 /// Set a DataLayout for this instance. If no data layout is specified then
306 /// the target's default data layout will be used.
307 SetterImpl &setDataLayout(Optional<DataLayout> DL) {
308 impl().DL = std::move(DL);
309 return impl();
310 }
311
312 /// Set an ObjectLinkingLayer creation function.
313 ///
314 /// If this method is not called, a default creation function will be used
315 /// that will construct an RTDyldObjectLinkingLayer.
316 SetterImpl &setObjectLinkingLayerCreator(
317 LLJITBuilderState::ObjectLinkingLayerCreator CreateObjectLinkingLayer) {
318 impl().CreateObjectLinkingLayer = std::move(CreateObjectLinkingLayer);
319 return impl();
320 }
321
322 /// Set a CompileFunctionCreator.
323 ///
324 /// If this method is not called, a default creation function wil be used
325 /// that will construct a basic IR compile function that is compatible with
326 /// the selected number of threads (SimpleCompiler for '0' compile threads,
327 /// ConcurrentIRCompiler otherwise).
328 SetterImpl &setCompileFunctionCreator(
329 LLJITBuilderState::CompileFunctionCreator CreateCompileFunction) {
330 impl().CreateCompileFunction = std::move(CreateCompileFunction);
331 return impl();
332 }
333
334 /// Set up an PlatformSetupFunction.
335 ///
336 /// If this method is not called then setUpGenericLLVMIRPlatform
337 /// will be used to configure the JIT's platform support.
338 SetterImpl &
339 setPlatformSetUp(LLJITBuilderState::PlatformSetupFunction SetUpPlatform) {
340 impl().SetUpPlatform = std::move(SetUpPlatform);
341 return impl();
342 }
343
344 /// Set the number of compile threads to use.
345 ///
346 /// If set to zero, compilation will be performed on the execution thread when
347 /// JITing in-process. If set to any other number N, a thread pool of N
348 /// threads will be created for compilation.
349 ///
350 /// If this method is not called, behavior will be as if it were called with
351 /// a zero argument.
352 SetterImpl &setNumCompileThreads(unsigned NumCompileThreads) {
353 impl().NumCompileThreads = NumCompileThreads;
354 return impl();
355 }
356
357 /// Create an instance of the JIT.
358 Expected<std::unique_ptr<JITType>> create() {
359 if (auto Err = impl().prepareForConstruction())
360 return std::move(Err);
361
362 Error Err = Error::success();
363 std::unique_ptr<JITType> J(new JITType(impl(), Err));
364 if (Err)
365 return std::move(Err);
366 return std::move(J);
367 }
368
369protected:
370 SetterImpl &impl() { return static_cast<SetterImpl &>(*this); }
371};

438/// Configure the LLJIT instance to use MachOPlatform support.
439///
440/// Warning: MachOPlatform *requires* that LLJIT be configured to use
441/// ObjectLinkingLayer (default on platforms supported by JITLink). If
442/// MachOPlatform is used with RTDyldObjectLinkingLayer it will result in
443/// undefined behavior).
444///
445/// MachOPlatform installs an ObjectLinkingLayer plugin to scrape initializers
446/// from the __mod_inits section. It also provides interposes for the dlfcn
447/// functions (dlopen, dlclose, dlsym, dlerror) that work for JITDylibs as
448/// well as regular libraries (JITDylibs will be preferenced, so make sure
449/// your JITDylib names do not shadow any real library paths).
