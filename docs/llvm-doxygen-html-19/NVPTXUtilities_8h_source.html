71// PTX ABI requires all scalar argument/return values to have
72// bit-size as a power of two of at least 32 bits.
73inline unsigned promoteScalarArgumentSize(unsigned size) {
74 if (size <= 32)
75 return 32;
76 else if (size <= 64)
77 return 64;
78 else
79 return size;
80}
