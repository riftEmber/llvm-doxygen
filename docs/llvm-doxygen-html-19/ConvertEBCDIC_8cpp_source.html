69ConverterEBCDIC::convertToEBCDIC(StringRef Source,
70 SmallVectorImpl<char> &Result) {
71 assert(Result.empty() && "Result must be empty!");
72 const unsigned char *Table = ISO88591ToIBM1047;
73 const unsigned char *Ptr =
74 reinterpret_cast<const unsigned char *>(Source.data());
75 size_t Length = Source.size();
76 Result.reserve(Length);
77 while (Length--) {
78 unsigned char Ch = *Ptr++;
79 // Handle UTF-8 2-byte-sequences in input.
80 if (Ch >= 128) {
81 // Only two-byte sequences can be decoded.
82 if (Ch != 0xc2 && Ch != 0xc3)
83 return std::make_error_code(std::errc::illegal_byte_sequence);
84 // Is buffer truncated?
85 if (!Length)
86 return std::make_error_code(std::errc::invalid_argument);
87 unsigned char Ch2 = *Ptr++;
88 // Is second byte well-formed?
89 if ((Ch2 & 0xc0) != 0x80)
90 return std::make_error_code(std::errc::illegal_byte_sequence);
91 Ch = Ch2 | (Ch << 6);
92 Length--;
93 }
94 // Translate the character.
95 Ch = Table[Ch];
96 Result.push_back(static_cast<char>(Ch));
97 }
98 return std::error_code();
99}

101void ConverterEBCDIC::convertToUTF8(StringRef Source,
102 SmallVectorImpl<char> &Result) {
103 assert(Result.empty() && "Result must be empty!");
104
105 const unsigned char *Table = IBM1047ToISO88591;
106 const unsigned char *Ptr =
107 reinterpret_cast<const unsigned char *>(Source.data());
108 size_t Length = Source.size();
109 Result.reserve(Length);
110 while (Length--) {
111 unsigned char Ch = *Ptr++;
112 // Translate the character.
113 Ch = Table[Ch];
114 // Handle UTF-8 2-byte-sequences in output.
115 if (Ch >= 128) {
116 // First byte prefixed with either 0xc2 or 0xc3.
117 Result.push_back(static_cast<char>(0xc0 | (Ch >> 6)));
118 // Second byte is either the same as the ASCII byte or ASCII byte -64.
119 Ch = Ch & 0xbf;
120 }
121 Result.push_back(static_cast<char>(Ch));
122 }
123}
