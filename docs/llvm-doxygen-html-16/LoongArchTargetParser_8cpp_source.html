38bool LoongArch::getArchFeatures(StringRef Arch,
39 std::vector<StringRef> &Features) {
40 for (const auto A : AllArchs) {
41 if (A.Name == Arch) {
42 for (const auto F : AllFeatures)
43 if ((A.Features & F.Kind) == F.Kind && F.Kind != FK_INVALID)
44 Features.push_back(F.Name);
