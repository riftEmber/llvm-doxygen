48LLVMComdatSelectionKind LLVMGetComdatSelectionKind(LLVMComdatRef C) {

51 return LLVMAnyComdatSelectionKind;
52 case Comdat::ExactMatch:
53 return LLVMExactMatchComdatSelectionKind;
54 case Comdat::Largest:
