29// CoroEarly pass marks every function that has coro.begin with a string
30// attribute "coroutine.presplit"="0". CoroSplit pass processes the coroutine
31// twice. First, it lets it go through complete IPO optimization pipeline as a
32// single function. It forces restart of the pipeline by inserting an indirect
33// call to an empty function "coro.devirt.trigger" which is devirtualized by
34// CoroElide pass that triggers a restart of the pipeline by CGPassManager.
35// When CoroSplit pass sees the same coroutine the second time, it splits it up,
36// adds coroutine subfunctions to the SCC to be processed by IPO pipeline.
37// Async lowering similarily triggers a restart of the pipeline after it has
38// split the coroutine.
39//
40// FIXME: Refactor these attributes as LLVM attributes instead of string
41// attributes since these attributes are already used outside LLVM's
42// coroutine module.
43// FIXME: Remove these values once we remove the Legacy PM.
44#define CORO_PRESPLIT_ATTR "coroutine.presplit"
45#define UNPREPARED_FOR_SPLIT "0"
46#define PREPARED_FOR_SPLIT "1"
47#define ASYNC_RESTART_AFTER_SPLIT "2"
48
49#define CORO_DEVIRT_TRIGGER_FN "coro.devirt.trigger"

76enum class ABI {
77 /// The "resume-switch" lowering, where there are separate resume and
78 /// destroy functions that are shared between all suspend points. The
79 /// coroutine frame implicitly stores the resume and destroy functions,
80 /// the current index, and any promise value.
81 Switch,
82
83 /// The "returned-continuation" lowering, where each suspend point creates a
84 /// single continuation function that is used for both resuming and
85 /// destroying. Does not support promises.
86 Retcon,
87
88 /// The "unique returned-continuation" lowering, where each suspend point
89 /// creates a single continuation function that is used for both resuming
90 /// and destroying. Does not support promises. The function is known to
91 /// suspend at most once during its execution, and the return value of
92 /// the continuation is void.
93 RetconOnce,
94
95 /// The "async continuation" lowering, where each suspend point creates a
96 /// single continuation function. The continuation function is available as an
97 /// intrinsic.
98 Async,
99};

211 FunctionType *getResumeFunctionType() const {
212 switch (ABI) {
213 case coro::ABI::Switch: {
214 auto *FnPtrTy = getSwitchResumePointerType();
215 return cast<FunctionType>(FnPtrTy->getPointerElementType());
216 }
217 case coro::ABI::Retcon:
218 case coro::ABI::RetconOnce:
219 return RetconLowering.ResumePrototype->getFunctionType();
220 case coro::ABI::Async:
221 // Not used. The function type depends on the active suspend.
222 return nullptr;
223 }
224
225 llvm_unreachable("Unknown coro::ABI enum");
226 }
227
228 ArrayRef<Type*> getRetconResultTypes() const {
229 assert(ABI == coro::ABI::Retcon ||
230 ABI == coro::ABI::RetconOnce);
231 auto FTy = CoroBegin->getFunction()->getFunctionType();
232
233 // The safety of all this is checked by checkWFRetconPrototype.
234 if (auto STy = dyn_cast<StructType>(FTy->getReturnType())) {
235 return STy->elements().slice(1);
236 } else {
237 return ArrayRef<Type*>();
238 }
239 }
240
241 ArrayRef<Type*> getRetconResumeTypes() const {
242 assert(ABI == coro::ABI::Retcon ||
243 ABI == coro::ABI::RetconOnce);
244
245 // The safety of all this is checked by checkWFRetconPrototype.
246 auto FTy = RetconLowering.ResumePrototype->getFunctionType();
247 return FTy->params().slice(1);
248 }

250 CallingConv::ID getResumeFunctionCC() const {
251 switch (ABI) {
252 case coro::ABI::Switch:
253 return CallingConv::Fast;
254
255 case coro::ABI::Retcon:
256 case coro::ABI::RetconOnce:
257 return RetconLowering.ResumePrototype->getCallingConv();
258 case coro::ABI::Async:
259 return AsyncLowering.AsyncCC;
260 }
261 llvm_unreachable("Unknown coro::ABI enum");
262 }
