30/// Class which can be overriden by targets to modify the
31/// mca::Instruction objects before the pipeline starts.
32/// A common usage of this class is to add immediate operands to certain
33/// instructions or to remove Defs/Uses from an instruction where the
34/// schedulinng model is incorrect.
35class InstrPostProcess {
36protected:
37 const MCSubtargetInfo &STI;
38 const MCInstrInfo &MCII;
39
40public:
41 InstrPostProcess(const MCSubtargetInfo &STI, const MCInstrInfo &MCII)
42 : STI(STI), MCII(MCII) {}
43
44 virtual ~InstrPostProcess() = default;

46 /// This method can be overriden by targets to modify the mca::Instruction
47 /// object after it has been lowered from the MCInst.
48 /// This is generally a less disruptive alternative to modifying the
49 /// scheduling model.
50 virtual void postProcessInstruction(std::unique_ptr<Instruction> &Inst,
51 const MCInst &MCI) {}
52};

54/// Class which can be overriden by targets to enforce instruction
55/// dependencies and behaviours that aren't expressed well enough
56/// within the scheduling model for mca to automatically simulate
57/// them properly.
58/// If you implement this class for your target, make sure to also implement
59/// a target specific InstrPostProcess class as well.
60class CustomBehaviour {
61protected:
62 const MCSubtargetInfo &STI;
63 const mca::SourceMgr &SrcMgr;
64 const MCInstrInfo &MCII;
65
66public:
67 CustomBehaviour(const MCSubtargetInfo &STI, const mca::SourceMgr &SrcMgr,
68 const MCInstrInfo &MCII)
69 : STI(STI), SrcMgr(SrcMgr), MCII(MCII) {}
70
71 virtual ~CustomBehaviour();
72
73 /// Before the llvm-mca pipeline dispatches an instruction, it first checks
74 /// for any register or resource dependencies / hazards. If it doesn't find
75 /// any, this method will be invoked to determine if there are any custom
76 /// hazards that the instruction needs to wait for.
77 /// The return value of this method is the number of cycles that the
78 /// instruction needs to wait for.
79 /// It's safe to underestimate the number of cycles to wait for since these
80 /// checks will be invoked again before the intruction gets dispatched.
81 /// However, it's not safe (accurate) to overestimate the number of cycles
82 /// to wait for since the instruction will wait for AT LEAST that number of
83 /// cycles before attempting to be dispatched again.
84 virtual unsigned checkCustomHazard(ArrayRef<InstRef> IssuedInst,
85 const InstRef &IR);
86
87 // Functions that target CBs can override to return a list of
88 // target specific Views that need to live within /lib/Target/ so that
89 // they can benefit from the target CB or from backend functionality that is
90 // not already exposed through MC-layer classes. Keep in mind that how this
91 // function is used is that the function is called within llvm-mca.cpp and
92 // then each unique_ptr<View> is passed into the PipelinePrinter::addView()
93 // function. This function will then std::move the View into its own vector of
94 // Views. So any CB that overrides this function needs to make sure that they
95 // are not relying on the current address or reference of the View
96 // unique_ptrs. If you do need the CB and View to be able to communicate with
97 // each other, consider giving the View a reference or pointer to the CB when
98 // the View is constructed. Then the View can query the CB for information
99 // when it needs it.
100 /// Return a vector of Views that will be added before all other Views.
101 virtual std::vector<std::unique_ptr<View>>
102 getStartViews(llvm::MCInstPrinter &IP, llvm::ArrayRef<llvm::MCInst> Insts);
103 /// Return a vector of Views that will be added after the InstructionInfoView.
104 virtual std::vector<std::unique_ptr<View>>
105 getPostInstrInfoViews(llvm::MCInstPrinter &IP,
106 llvm::ArrayRef<llvm::MCInst> Insts);
107 /// Return a vector of Views that will be added after all other Views.
108 virtual std::vector<std::unique_ptr<View>>
109 getEndViews(llvm::MCInstPrinter &IP, llvm::ArrayRef<llvm::MCInst> Insts);
110};
