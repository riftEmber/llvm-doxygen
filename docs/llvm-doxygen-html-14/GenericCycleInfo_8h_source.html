8///
9/// \file
10/// \brief Find all cycles in a control-flow graph, including irreducible loops.
11///
12/// See docs/CycleTerminology.rst for a formal definition of cycles.
13///
14/// Briefly:
15/// - A cycle is a generalization of a loop which can represent
16/// irreducible control flow.
17/// - Cycles identified in a program are implementation defined,
18/// depending on the DFS traversal chosen.
19/// - Cycles are well-nested, and form a forest with a parent-child
20/// relationship.
21/// - In any choice of DFS, every natural loop L is represented by a
22/// unique cycle C which is a superset of L.
23/// - In the absence of irreducible control flow, the cycles are
24/// exactly the natural loops in the program.
25///
26//===----------------------------------------------------------------------===//

47/// A possibly irreducible generalization of a \ref Loop.
48template <typename ContextT> class GenericCycle {
49public:
50 using BlockT = typename ContextT::BlockT;
51 using FunctionT = typename ContextT::FunctionT;
52 template <typename> friend class GenericCycleInfo;
53 template <typename> friend class GenericCycleInfoCompute;
54
55private:
56 /// The parent cycle. Is null for the root "cycle". Top-level cycles point
57 /// at the root.
58 GenericCycle *ParentCycle = nullptr;
59
60 /// The entry block(s) of the cycle. The header is the only entry if
61 /// this is a loop. Is empty for the root "cycle", to avoid
62 /// unnecessary memory use.
63 SmallVector<BlockT *, 1> Entries;
64
65 /// Child cycles, if any.
66 std::vector<std::unique_ptr<GenericCycle>> Children;
67
68 /// Basic blocks that are contained in the cycle, including entry blocks,
69 /// and including blocks that are part of a child cycle.
70 std::vector<BlockT *> Blocks;
71
72 /// Depth of the cycle in the tree. The root "cycle" is at depth 0.
73 ///
74 /// \note Depths are not necessarily contiguous. However, child loops always
75 /// have strictly greater depth than their parents, and sibling loops
76 /// always have the same depth.
77 unsigned Depth = 0;
78
79 void clear() {
80 Entries.clear();
81 Children.clear();
82 Blocks.clear();
83 Depth = 0;
84 ParentCycle = nullptr;
85 }
86
87 void appendEntry(BlockT *Block) { Entries.push_back(Block); }
88 void appendBlock(BlockT *Block) { Blocks.push_back(Block); }
89
90 GenericCycle(const GenericCycle &) = delete;
91 GenericCycle &operator=(const GenericCycle &) = delete;
92 GenericCycle(GenericCycle &&Rhs) = delete;
93 GenericCycle &operator=(GenericCycle &&Rhs) = delete;
94
95public:
96 GenericCycle() = default;
97
98 /// \brief Whether the cycle is a natural loop.
99 bool isReducible() const { return Entries.size() == 1; }
100
101 BlockT *getHeader() const { return Entries[0]; }
102
103 /// \brief Return whether \p Block is an entry block of the cycle.
104 bool isEntry(BlockT *Block) const { return is_contained(Entries, Block); }
105
106 /// \brief Return whether \p Block is contained in the cycle.
107 bool contains(const BlockT *Block) const {
108 return is_contained(Blocks, Block);
109 }
110
111 /// \brief Returns true iff this cycle contains \p C.
112 ///
113 /// Note: Non-strict containment check, i.e. returns true if C is the
114 /// same cycle.
115 bool contains(const GenericCycle *C) const;
116
117 const GenericCycle *getParentCycle() const { return ParentCycle; }
118 GenericCycle *getParentCycle() { return ParentCycle; }
119 unsigned getDepth() const { return Depth; }
120
121 /// Return all of the successor blocks of this cycle.
122 ///
123 /// These are the blocks _outside of the current cycle_ which are
124 /// branched to.
125 void getExitBlocks(SmallVectorImpl<BlockT *> &TmpStorage) const;
126
127 /// Iteration over child cycles.
128 //@{
129 using const_child_iterator_base =
130 typename std::vector<std::unique_ptr<GenericCycle>>::const_iterator;
131 struct const_child_iterator
132 : iterator_adaptor_base<const_child_iterator, const_child_iterator_base> {
133 using Base =
134 iterator_adaptor_base<const_child_iterator, const_child_iterator_base>;
135
136 const_child_iterator() = default;
137 explicit const_child_iterator(const_child_iterator_base I) : Base(I) {}
138
139 const const_child_iterator_base &wrapped() { return Base::wrapped(); }
140 GenericCycle *operator*() const { return Base::I->get(); }
141 };
142
143 const_child_iterator child_begin() const {
144 return const_child_iterator{Children.begin()};
145 }
146 const_child_iterator child_end() const {
147 return const_child_iterator{Children.end()};
148 }
149 size_t getNumChildren() const { return Children.size(); }
150 iterator_range<const_child_iterator> children() const {
151 return llvm::make_range(const_child_iterator{Children.begin()},
152 const_child_iterator{Children.end()});
153 }
154 //@}
155
156 /// Iteration over blocks in the cycle (including entry blocks).
157 //@{
158 using const_block_iterator = typename std::vector<BlockT *>::const_iterator;
159
160 const_block_iterator block_begin() const {
161 return const_block_iterator{Blocks.begin()};
162 }
163 const_block_iterator block_end() const {
164 return const_block_iterator{Blocks.end()};
165 }
166 size_t getNumBlocks() const { return Blocks.size(); }
167 iterator_range<const_block_iterator> blocks() const {
168 return llvm::make_range(block_begin(), block_end());
169 }
170 //@}
171
172 /// Iteration over entry blocks.
173 //@{
174 using const_entry_iterator =
175 typename SmallVectorImpl<BlockT *>::const_iterator;
176
177 size_t getNumEntries() const { return Entries.size(); }
178 iterator_range<const_entry_iterator> entries() const {
179 return llvm::make_range(Entries.begin(), Entries.end());
180 }
181
182 Printable printEntries(const ContextT &Ctx) const {
183 return Printable([this, &Ctx](raw_ostream &Out) {
184 bool First = true;
185 for (auto *Entry : Entries) {
186 if (!First)
187 Out << ' ';
188 First = false;
189 Out << Ctx.print(Entry);
190 }
191 });
192 }
193
194 Printable print(const ContextT &Ctx) const {
195 return Printable([this, &Ctx](raw_ostream &Out) {
196 Out << "depth=" << Depth << ": entries(" << printEntries(Ctx) << ')';
197
198 for (auto *Block : Blocks) {
199 if (isEntry(Block))
200 continue;
201
202 Out << ' ' << Ctx.print(Block);
203 }
204 });
205 }
206};

208/// \brief Cycle information for a function.
209template <typename ContextT> class GenericCycleInfo {
210public:
211 using BlockT = typename ContextT::BlockT;
212 using CycleT = GenericCycle<ContextT>;
213 using FunctionT = typename ContextT::FunctionT;
214 template <typename> friend class GenericCycle;
215 template <typename> friend class GenericCycleInfoCompute;
216
217private:
218 ContextT Context;
219
220 /// Map basic blocks to their inner-most containing loop.
221 DenseMap<BlockT *, CycleT *> BlockMap;
222
223 /// Outermost cycles discovered by any DFS.
224 ///
225 /// Note: The implementation treats the nullptr as the parent of
226 /// every top-level cycle. See \ref contains for an example.
227 std::vector<std::unique_ptr<CycleT>> TopLevelCycles;
228
229public:
230 GenericCycleInfo() = default;
231 GenericCycleInfo(GenericCycleInfo &&) = default;
232 GenericCycleInfo &operator=(GenericCycleInfo &&) = default;
233
234 void clear();
235 void compute(FunctionT &F);
236
237 FunctionT *getFunction() const { return Context.getFunction(); }
238 const ContextT &getSSAContext() const { return Context; }
239
240 CycleT *getCycle(const BlockT *Block) const;
241 CycleT *getTopLevelParentCycle(const BlockT *Block) const;
242
243 /// Move \p Child to \p NewParent by manipulating Children vectors.
244 ///
245 /// Note: This is an incomplete operation that does not update the
246 /// list of blocks in the new parent or the depth of the subtree.
247 void moveToNewParent(CycleT *NewParent, CycleT *Child);
248
249 /// Methods for debug and self-test.
250 //@{
251 bool validateTree() const;
252 void print(raw_ostream &Out) const;
253 void dump() const { print(dbgs()); }
254 //@}

283 const_toplevel_iterator{TopLevelCycles.end()});
284 }
285 //@}
286};
