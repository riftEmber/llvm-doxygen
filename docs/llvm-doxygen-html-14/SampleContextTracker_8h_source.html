88// Profile tracker that manages profiles and its associated context. It
89// provides interfaces used by sample profile loader to query context profile or
90// base profile for given function or location; it also manages context tree
91// manipulation that is needed to accommodate inline decisions so we have
92// accurate post-inline profile for functions. Internally context profiles
93// are organized in a trie, with each node representing profile for specific
94// calling context and the context is identified by path from root to the node.
95class SampleContextTracker {
96public:
97 struct ProfileComparer {
98 bool operator()(FunctionSamples *A, FunctionSamples *B) const {
99 // Sort function profiles by the number of total samples and their
100 // contexts.
101 if (A->getTotalSamples() == B->getTotalSamples())
102 return A->getContext() < B->getContext();
103 return A->getTotalSamples() > B->getTotalSamples();
104 }
105 };
106
107 // Keep profiles of a function sorted so that they will be processed/promoted
108 // deterministically.
109 using ContextSamplesTy = std::set<FunctionSamples *, ProfileComparer>;
110
111 SampleContextTracker(SampleProfileMap &Profiles,
112 const DenseMap<uint64_t, StringRef> *GUIDToFuncNameMap);
113 // Query context profile for a specific callee with given name at a given
114 // call-site. The full context is identified by location of call instruction.
115 FunctionSamples *getCalleeContextSamplesFor(const CallBase &Inst,
116 StringRef CalleeName);
117 // Get samples for indirect call targets for call site at given location.
