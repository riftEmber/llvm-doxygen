124 assert(F && "A concrete function must be provided to this routine.");
125
126 // FIXME: These should almost certainly not be handled here, and instead
127 // handled with the help of TLI or the target itself. This was largely
128 // ported from existing analysis heuristics here so that such refactorings
129 // can take place in the future.
130
131 if (F->isIntrinsic())
132 return false;
133
134 if (F->hasLocalLinkage() || !F->hasName())

140 if (Name == "copysign" || Name == "copysignf" || Name == "copysignl" ||
141 Name == "fabs" || Name == "fabsf" || Name == "fabsl" || Name == "sin" ||
142 Name == "fmin" || Name == "fminf" || Name == "fminl" ||
143 Name == "fmax" || Name == "fmaxf" || Name == "fmaxl" ||

156 return true;

497 InstructionCost getCastInstrCost(unsigned Opcode, Type *Dst, Type *Src,
498 TTI::CastContextHint CCH,
499 TTI::TargetCostKind CostKind,
500 const Instruction *I) const {
501 switch (Opcode) {
502 default:
503 break;
504 case Instruction::IntToPtr: {
505 unsigned SrcSize = Src->getScalarSizeInBits();
506 if (DL.isLegalInteger(SrcSize) &&
507 SrcSize <= DL.getPointerTypeSizeInBits(Dst))
508 return 0;
509 break;
510 }
511 case Instruction::PtrToInt: {
512 unsigned DstSize = Dst->getScalarSizeInBits();
513 if (DL.isLegalInteger(DstSize) &&
514 DstSize >= DL.getPointerTypeSizeInBits(Src))
515 return 0;
516 break;
517 }
518 case Instruction::BitCast:
519 if (Dst == Src || (Dst->isPointerTy() && Src->isPointerTy()))
520 // Identity and pointer-to-pointer casts are free.
521 return 0;
522 break;
523 case Instruction::Trunc: {
524 // trunc to a native type is free (assuming the target has compare and
525 // shift-right of the same width).
526 TypeSize DstSize = DL.getTypeSizeInBits(Dst);
527 if (!DstSize.isScalable() && DL.isLegalInteger(DstSize.getFixedSize()))
528 return 0;
529 break;
530 }
531 }
532 return 1;
533 }

804 // Obtain the minimum required size to hold the value (without the sign)
805 // In case of a vector it returns the min required size for one element.
806 unsigned minRequiredElementSize(const Value *Val, bool &isSigned) const {
807 if (isa<ConstantDataVector>(Val) || isa<ConstantVector>(Val)) {
808 const auto *VectorValue = cast<Constant>(Val);
809
810 // In case of a vector need to pick the max between the min
811 // required size for each element
812 auto *VT = cast<FixedVectorType>(Val->getType());
813
814 // Assume unsigned elements
815 isSigned = false;
816
817 // The max required size is the size of the vector element type
818 unsigned MaxRequiredSize =
819 VT->getElementType()->getPrimitiveSizeInBits().getFixedSize();
820
821 unsigned MinRequiredSize = 0;
822 for (unsigned i = 0, e = VT->getNumElements(); i < e; ++i) {
823 if (auto *IntElement =
824 dyn_cast<ConstantInt>(VectorValue->getAggregateElement(i))) {
825 bool signedElement = IntElement->getValue().isNegative();
826 // Get the element min required size.
827 unsigned ElementMinRequiredSize =
828 IntElement->getValue().getMinSignedBits() - 1;
829 // In case one element is signed then all the vector is signed.
830 isSigned |= signedElement;
831 // Save the max required bit size between all the elements.
832 MinRequiredSize = std::max(MinRequiredSize, ElementMinRequiredSize);
833 } else {
834 // not an int constant element
835 return MaxRequiredSize;
836 }
837 }
838 return MinRequiredSize;
839 }
840
841 if (const auto *CI = dyn_cast<ConstantInt>(Val)) {
842 isSigned = CI->getValue().isNegative();
843 return CI->getValue().getMinSignedBits() - 1;
844 }
845
846 if (const auto *Cast = dyn_cast<SExtInst>(Val)) {
847 isSigned = true;
848 return Cast->getSrcTy()->getScalarSizeInBits() - 1;
849 }
850
851 if (const auto *Cast = dyn_cast<ZExtInst>(Val)) {
852 isSigned = false;
853 return Cast->getSrcTy()->getScalarSizeInBits();
854 }
855
856 isSigned = false;
857 return Val->getType()->getScalarSizeInBits();
858 }

898 InstructionCost getGEPCost(Type *PointeeType, const Value *Ptr,
899 ArrayRef<const Value *> Operands,
900 TTI::TargetCostKind CostKind) {
901 assert(PointeeType && Ptr && "can't get GEPCost of nullptr");
902 assert(cast<PointerType>(Ptr->getType()->getScalarType())
903 ->isOpaqueOrPointeeTypeMatches(PointeeType) &&
904 "explicit pointee type doesn't match operand's pointee type");
905 auto *BaseGV = dyn_cast<GlobalValue>(Ptr->stripPointerCasts());
906 bool HasBaseReg = (BaseGV == nullptr);
907
908 auto PtrSizeBits = DL.getPointerTypeSizeInBits(Ptr->getType());
909 APInt BaseOffset(PtrSizeBits, 0);
910 int64_t Scale = 0;
911
912 auto GTI = gep_type_begin(PointeeType, Operands);
913 Type *TargetType = nullptr;
914
915 // Handle the case where the GEP instruction has a single operand,
916 // the basis, therefore TargetType is a nullptr.
917 if (Operands.empty())
918 return !BaseGV ? TTI::TCC_Free : TTI::TCC_Basic;
919
920 for (auto I = Operands.begin(); I != Operands.end(); ++I, ++GTI) {
921 TargetType = GTI.getIndexedType();
922 // We assume that the cost of Scalar GEP with constant index and the
923 // cost of Vector GEP with splat constant index are the same.
924 const ConstantInt *ConstIdx = dyn_cast<ConstantInt>(*I);
925 if (!ConstIdx)
926 if (auto Splat = getSplatValue(*I))
927 ConstIdx = dyn_cast<ConstantInt>(Splat);
928 if (StructType *STy = GTI.getStructTypeOrNull()) {
929 // For structures the index is always splat or scalar constant
930 assert(ConstIdx && "Unexpected GEP index");
931 uint64_t Field = ConstIdx->getZExtValue();
932 BaseOffset += DL.getStructLayout(STy)->getElementOffset(Field);
933 } else {
934 // If this operand is a scalable type, bail out early.
935 // TODO: handle scalable vectors
936 if (isa<ScalableVectorType>(TargetType))
937 return TTI::TCC_Basic;
938 int64_t ElementSize =
939 DL.getTypeAllocSize(GTI.getIndexedType()).getFixedSize();
940 if (ConstIdx) {
941 BaseOffset +=
942 ConstIdx->getValue().sextOrTrunc(PtrSizeBits) * ElementSize;
943 } else {
944 // Needs scale register.
945 if (Scale != 0)
946 // No addressing mode takes two scale registers.
947 return TTI::TCC_Basic;
948 Scale = ElementSize;
949 }
950 }
951 }
952
953 if (static_cast<T *>(this)->isLegalAddressingMode(
954 TargetType, const_cast<GlobalValue *>(BaseGV),
955 BaseOffset.sextOrTrunc(64).getSExtValue(), HasBaseReg, Scale,
956 Ptr->getType()->getPointerAddressSpace()))
957 return TTI::TCC_Free;
958 return TTI::TCC_Basic;
959 }

961 InstructionCost getUserCost(const User *U, ArrayRef<const Value *> Operands,
962 TTI::TargetCostKind CostKind) {
963 auto *TargetTTI = static_cast<T *>(this);
964 // Handle non-intrinsic calls, invokes, and callbr.
965 // FIXME: Unlikely to be true for anything but CodeSize.
966 auto *CB = dyn_cast<CallBase>(U);
967 if (CB && !isa<IntrinsicInst>(U)) {
968 if (const Function *F = CB->getCalledFunction()) {
969 if (!TargetTTI->isLoweredToCall(F))
970 return TTI::TCC_Basic; // Give a basic cost if it will be lowered
971
972 return TTI::TCC_Basic * (F->getFunctionType()->getNumParams() + 1);
973 }
974 // For indirect or other calls, scale cost by number of arguments.
975 return TTI::TCC_Basic * (CB->arg_size() + 1);

977
978 Type *Ty = U->getType();

980 U->getNumOperands() == 1 ? U->getOperand(0)->getType() : nullptr;
981 unsigned Opcode = Operator::getOpcode(U);

992 case Instruction::Br:
993 case Instruction::Ret:
994 case Instruction::PHI:
995 case Instruction::Switch:

997 case Instruction::ExtractValue:
998 case Instruction::Freeze:

1000 case Instruction::Alloca:

1002 return TTI::TCC_Free;

1011 case Instruction::FAdd:
1012 case Instruction::Sub:
1013 case Instruction::FSub:
1014 case Instruction::Mul:
1015 case Instruction::FMul:
1016 case Instruction::UDiv:
1017 case Instruction::SDiv:
1018 case Instruction::FDiv:
1019 case Instruction::URem:
1020 case Instruction::SRem:
1021 case Instruction::FRem:
1022 case Instruction::Shl:
1023 case Instruction::LShr:
1024 case Instruction::AShr:
1025 case Instruction::And:
1026 case Instruction::Or:
1027 case Instruction::Xor:
1028 case Instruction::FNeg: {
