24Optional<std::string>
25LocalDIFetcher::fetchBuildID(ArrayRef<uint8_t> BuildID) const {
26 auto GetDebugPath = [&](StringRef Directory) {
27 SmallString<128> Path{Directory};
28 sys::path::append(Path, ".build-id",
29 llvm::toHex(BuildID[0], /*LowerCase=*/true),
30 llvm::toHex(BuildID.slice(1), /*LowerCase=*/true));
31 Path += ".debug";
32 return Path;
33 };
34 if (DebugFileDirectory.empty()) {
35 SmallString<128> Path = GetDebugPath(
36#if defined(__NetBSD__)
37 // Try /usr/libdata/debug/.build-id/../...
38 "/usr/libdata/debug"
39#else
40 // Try /usr/lib/debug/.build-id/../...
41 "/usr/lib/debug"
42#endif
43 );
44 if (llvm::sys::fs::exists(Path))
45 return std::string(Path);
46 } else {
47 for (const auto &Directory : DebugFileDirectory) {
48 // Try <debug-file-directory>/.build-id/../...
49 SmallString<128> Path = GetDebugPath(Directory);
50 if (llvm::sys::fs::exists(Path))
51 return std::string(Path);
52 }
53 }
54 return None;
55}
