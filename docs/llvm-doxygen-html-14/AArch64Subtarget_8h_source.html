643 /// Return whether FrameLowering should always set the "extended frame
644 /// present" bit in FP, or set it based on a symbol in the runtime.
645 bool swiftAsyncContextIsDynamicallySet() const {
646 // Older OS versions (particularly system unwinders) are confused by the
647 // Swift extended frame, so when building code that might be run on them we
648 // must dynamically query the concurrency library to determine whether
649 // extended frames should be flagged as present.
650 const Triple &TT = getTargetTriple();
651
652 unsigned Major = TT.getOSVersion().getMajor();
653 switch(TT.getOS()) {
654 default:
655 return false;
656 case Triple::IOS:
657 case Triple::TvOS:
658 return Major < 15;
659 case Triple::WatchOS:
660 return Major < 8;
661 case Triple::MacOSX:
662 case Triple::Darwin:
663 return Major < 12;
664 }
665 }
