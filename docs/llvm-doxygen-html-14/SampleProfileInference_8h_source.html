83/// Sample profile inference pass.
84template <typename BT> class SampleProfileInference {
85public:
86 using BasicBlockT = typename afdo_detail::TypeMap<BT>::BasicBlockT;
87 using FunctionT = typename afdo_detail::TypeMap<BT>::FunctionT;
88 using Edge = std::pair<const BasicBlockT *, const BasicBlockT *>;
89 using BlockWeightMap = DenseMap<const BasicBlockT *, uint64_t>;
90 using EdgeWeightMap = DenseMap<Edge, uint64_t>;
91 using BlockEdgeMap =
92 DenseMap<const BasicBlockT *, SmallVector<const BasicBlockT *, 8>>;
93
94 SampleProfileInference(FunctionT &F, BlockEdgeMap &Successors,
95 BlockWeightMap &SampleBlockWeights)
96 : F(F), Successors(Successors), SampleBlockWeights(SampleBlockWeights) {}
97
98 /// Apply the profile inference algorithm for a given function
99 void apply(BlockWeightMap &BlockWeights, EdgeWeightMap &EdgeWeights);
100
101private:
102 /// Try to infer branch probabilities mimicking implementation of
103 /// BranchProbabilityInfo. Unlikely taken branches are marked so that the
104 /// inference algorithm can avoid sending flow along corresponding edges.
105 void findUnlikelyJumps(const std::vector<const BasicBlockT *> &BasicBlocks,
106 BlockEdgeMap &Successors, FlowFunction &Func);
107
108 /// Determine whether the block is an exit in the CFG.
109 bool isExit(const BasicBlockT *BB);
110
111 /// Function.
112 const FunctionT &F;
113
114 /// Successors for each basic block in the CFG.
115 BlockEdgeMap &Successors;
116
117 /// Map basic blocks to their sampled weights.
118 BlockWeightMap &SampleBlockWeights;
119};

121template <typename BT>
122void SampleProfileInference<BT>::apply(BlockWeightMap &BlockWeights,
123 EdgeWeightMap &EdgeWeights) {
124 // Find all forwards reachable blocks which the inference algorithm will be
125 // applied on.
126 df_iterator_default_set<const BasicBlockT *> Reachable;
127 for (auto *BB : depth_first_ext(&F, Reachable))
128 (void)BB /* Mark all reachable blocks */;
129
130 // Find all backwards reachable blocks which the inference algorithm will be
131 // applied on.
132 df_iterator_default_set<const BasicBlockT *> InverseReachable;
133 for (const auto &BB : F) {
134 // An exit block is a block without any successors.
135 if (isExit(&BB)) {
136 for (auto *RBB : inverse_depth_first_ext(&BB, InverseReachable))
137 (void)RBB;
138 }
139 }
140
141 // Keep a stable order for reachable blocks
142 DenseMap<const BasicBlockT *, uint64_t> BlockIndex;
143 std::vector<const BasicBlockT *> BasicBlocks;
144 BlockIndex.reserve(Reachable.size());
145 BasicBlocks.reserve(Reachable.size());
146 for (const auto &BB : F) {
147 if (Reachable.count(&BB) && InverseReachable.count(&BB)) {
148 BlockIndex[&BB] = BasicBlocks.size();
149 BasicBlocks.push_back(&BB);
150 }
151 }
152
153 BlockWeights.clear();
154 EdgeWeights.clear();
155 bool HasSamples = false;
156 for (const auto *BB : BasicBlocks) {
157 auto It = SampleBlockWeights.find(BB);
158 if (It != SampleBlockWeights.end() && It->second > 0) {
159 HasSamples = true;
160 BlockWeights[BB] = It->second;
161 }
162 }
163 // Quit early for functions with a single block or ones w/o samples
164 if (BasicBlocks.size() <= 1 || !HasSamples) {
165 return;
166 }
167
168 // Create necessary objects
169 FlowFunction Func;
170 Func.Blocks.reserve(BasicBlocks.size());
171 // Create FlowBlocks
172 for (const auto *BB : BasicBlocks) {
173 FlowBlock Block;
174 if (SampleBlockWeights.find(BB) != SampleBlockWeights.end()) {
175 Block.UnknownWeight = false;
176 Block.Weight = SampleBlockWeights[BB];
177 } else {
178 Block.UnknownWeight = true;
179 Block.Weight = 0;
180 }
181 Block.Index = Func.Blocks.size();
182 Func.Blocks.push_back(Block);
183 }
184 // Create FlowEdges
185 for (const auto *BB : BasicBlocks) {
186 for (auto *Succ : Successors[BB]) {
187 if (!BlockIndex.count(Succ))
188 continue;
189 FlowJump Jump;
190 Jump.Source = BlockIndex[BB];
191 Jump.Target = BlockIndex[Succ];
192 Func.Jumps.push_back(Jump);
193 if (BB == Succ) {
194 Func.Blocks[BlockIndex[BB]].HasSelfEdge = true;
195 }
196 }
197 }
198 for (auto &Jump : Func.Jumps) {
199 Func.Blocks[Jump.Source].SuccJumps.push_back(&Jump);
200 Func.Blocks[Jump.Target].PredJumps.push_back(&Jump);
201 }
202
203 // Try to infer probabilities of jumps based on the content of basic block
204 findUnlikelyJumps(BasicBlocks, Successors, Func);
205
206 // Find the entry block
207 for (size_t I = 0; I < Func.Blocks.size(); I++) {
208 if (Func.Blocks[I].isEntry()) {
209 Func.Entry = I;
210 break;
211 }
212 }
213
214 // Create and apply the inference network model.
215 applyFlowInference(Func);
216
217 // Extract the resulting weights from the control flow
218 // All weights are increased by one to avoid propagation errors introduced by
219 // zero weights.
220 for (const auto *BB : BasicBlocks) {
221 BlockWeights[BB] = Func.Blocks[BlockIndex[BB]].Flow;
222 }
223 for (auto &Jump : Func.Jumps) {
224 Edge E = std::make_pair(BasicBlocks[Jump.Source], BasicBlocks[Jump.Target]);
225 EdgeWeights[E] = Jump.Flow;
226 }
227
228#ifndef NDEBUG
229 // Unreachable blocks and edges should not have a weight.
230 for (auto &I : BlockWeights) {
231 assert(Reachable.contains(I.first));
232 assert(InverseReachable.contains(I.first));
233 }
234 for (auto &I : EdgeWeights) {
235 assert(Reachable.contains(I.first.first) &&
236 Reachable.contains(I.first.second));
237 assert(InverseReachable.contains(I.first.first) &&
238 InverseReachable.contains(I.first.second));
239 }
240#endif
241}

244inline void SampleProfileInference<BT>::findUnlikelyJumps(
245 const std::vector<const BasicBlockT *> &BasicBlocks,
246 BlockEdgeMap &Successors, FlowFunction &Func) {}

248template <>
249inline void SampleProfileInference<BasicBlock>::findUnlikelyJumps(
250 const std::vector<const BasicBlockT *> &BasicBlocks,
251 BlockEdgeMap &Successors, FlowFunction &Func) {
252 for (auto &Jump : Func.Jumps) {
253 const auto *BB = BasicBlocks[Jump.Source];
254 const auto *Succ = BasicBlocks[Jump.Target];
255 const Instruction *TI = BB->getTerminator();
256 // Check if a block ends with InvokeInst and mark non-taken branch unlikely.
257 // In that case block Succ should be a landing pad
258 if (Successors[BB].size() == 2 && Successors[BB].back() == Succ) {
259 if (isa<InvokeInst>(TI)) {
260 Jump.IsUnlikely = true;
261 }
262 }
263 const Instruction *SuccTI = Succ->getTerminator();
264 // Check if the target block contains UnreachableInst and mark it unlikely
265 if (SuccTI->getNumSuccessors() == 0) {
266 if (isa<UnreachableInst>(SuccTI)) {
267 Jump.IsUnlikely = true;
268 }
269 }
270 }
