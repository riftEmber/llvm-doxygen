21bool convertYAML(yaml::Input &YIn, raw_ostream &Out, ErrorHandler ErrHandler,
22 unsigned DocNum, uint64_t MaxSize) {
23 unsigned CurDocNum = 0;
24 do {
25 if (++CurDocNum != DocNum)
26 continue;
27
28 yaml::YamlObjectFile Doc;
29 YIn >> Doc;
30 if (std::error_code EC = YIn.error()) {
31 ErrHandler("failed to parse YAML input: " + EC.message());
32 return false;
33 }
34
35 if (Doc.Arch)
36 return yaml2archive(*Doc.Arch, Out, ErrHandler);
37 if (Doc.Elf)
38 return yaml2elf(*Doc.Elf, Out, ErrHandler, MaxSize);
39 if (Doc.Coff)
40 return yaml2coff(*Doc.Coff, Out, ErrHandler);
41 if (Doc.MachO || Doc.FatMachO)
42 return yaml2macho(Doc, Out, ErrHandler);
43 if (Doc.Minidump)
44 return yaml2minidump(*Doc.Minidump, Out, ErrHandler);
45 if (Doc.Wasm)
46 return yaml2wasm(*Doc.Wasm, Out, ErrHandler);
47 if (Doc.Xcoff)
48 return yaml2xcoff(*Doc.Xcoff, Out, ErrHandler);
49
50 ErrHandler("unknown document type");
51 return false;
52
53 } while (YIn.nextDocument());
54
55 ErrHandler("cannot find the " + Twine(DocNum) +
56 getOrdinalSuffix(DocNum).data() + " document");
57 return false;
58}

60std::unique_ptr<object::ObjectFile>
61yaml2ObjectFile(SmallVectorImpl<char> &Storage, StringRef Yaml,
62 ErrorHandler ErrHandler) {
63 Storage.clear();
64 raw_svector_ostream OS(Storage);
65
66 yaml::Input YIn(Yaml);
67 if (!convertYAML(YIn, OS, ErrHandler))
68 return {};
69
70 Expected<std::unique_ptr<object::ObjectFile>> ObjOrErr =
71 object::ObjectFile::createObjectFile(
72 MemoryBufferRef(OS.str(), "YamlObject"));
73 if (ObjOrErr)
74 return std::move(*ObjOrErr);
75
76 ErrHandler(toString(ObjOrErr.takeError()));
77 return {};
78}
