49/// Analysis pass providing a never-invalidated alias analysis result.
50class AMDGPUAA : public AnalysisInfoMixin<AMDGPUAA> {
51 friend AnalysisInfoMixin<AMDGPUAA>;

53 static AnalysisKey Key;
54
55public:
