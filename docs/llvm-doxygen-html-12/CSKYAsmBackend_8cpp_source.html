48bool CSKYAsmBackend::writeNopData(raw_ostream &OS, uint64_t Count) const {
49 if (Count % 2)
50 return false;
51
52 // MOV32 r0, r0
53 while (Count >= 4) {
54 OS.write("\xc4\x00\x48\x20", 4);
55 Count -= 4;
56 }
57 // MOV16 r0, r0
58 if (Count)
59 OS.write("\x6c\x03", 2);
60
61 return true;
62}
