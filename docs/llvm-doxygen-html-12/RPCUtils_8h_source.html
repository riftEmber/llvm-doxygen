314// RespondHelper specialization for functions that support error returns.
315template <> class RespondHelper<true> {
316public:
317 // Send Expected<T>.
318 template <typename WireRetT, typename HandlerRetT, typename ChannelT,
319 typename FunctionIdT, typename SequenceNumberT>
320 static Error sendResult(ChannelT &C, const FunctionIdT &ResponseId,
321 SequenceNumberT SeqNo,
322 Expected<HandlerRetT> ResultOrErr) {
323 if (!ResultOrErr && ResultOrErr.template errorIsA<RPCFatalError>())
324 return ResultOrErr.takeError();
325
326 // Open the response message.
327 if (auto Err = C.startSendMessage(ResponseId, SeqNo))
328 return Err;
329
330 // Serialize the result.
331 if (auto Err =
332 SerializationTraits<ChannelT, WireRetT, Expected<HandlerRetT>>::
333 serialize(C, std::move(ResultOrErr)))
334 return Err;
335
336 // Close the response message.
337 if (auto Err = C.endSendMessage())
338 return Err;
339 return C.send();
340 }
341
342 template <typename ChannelT, typename FunctionIdT, typename SequenceNumberT>
343 static Error sendResult(ChannelT &C, const FunctionIdT &ResponseId,
344 SequenceNumberT SeqNo, Error Err) {
345 if (Err && Err.isA<RPCFatalError>())
346 return Err;
347 if (auto Err2 = C.startSendMessage(ResponseId, SeqNo))
348 return Err2;
349 if (auto Err2 = serializeSeq(C, std::move(Err)))
350 return Err2;
351 if (auto Err2 = C.endSendMessage())
352 return Err2;
353 return C.send();
354 }
355};

739 // Handle the result by deserializing it from the channel then passing it
740 // to the user defined handler.
741 Error handleResponse(ChannelT &C) override {
742 Error Result = Error::success();
743 if (auto Err = SerializationTraits<ChannelT, Error, Error>::deserialize(
744 C, Result)) {
745 consumeError(std::move(Result));
746 return Err;
747 }
748 if (auto Err = C.endReceiveMessage()) {
749 consumeError(std::move(Result));
750 return Err;
751 }
752 return Handler(std::move(Result));
753 }
754
755 // Abandon this response by calling the handler with an 'abandoned response'
756 // error.
757 void abandon() override {
758 if (auto Err = Handler(this->createAbandonedResponseError())) {
759 // Handlers should not fail when passed an abandoned response error.
760 report_fatal_error(std::move(Err));
761 }
762 }
763
764private:
765 HandlerT Handler;
766};

995 /// Append a call Func, does not call send on the channel.
996 /// The first argument specifies a user-defined handler to be run when the
997 /// function returns. The handler should take an Expected<Func::ReturnType>,
998 /// or an Error (if Func::ReturnType is void). The handler will be called
999 /// with an error if the return value is abandoned due to a channel error.
1000 template <typename Func, typename HandlerT, typename... ArgTs>
1001 Error appendCallAsync(HandlerT Handler, const ArgTs &...Args) {
1002
1003 static_assert(
1004 detail::RPCArgTypeCheck<CanSerializeCheck, typename Func::Type,
1005 void(ArgTs...)>::value,
1006 "");
1007
1008 // Look up the function ID.
1009 FunctionIdT FnId;
1010 if (auto FnIdOrErr = getRemoteFunctionId<Func>(LazyAutoNegotiation, false))
1011 FnId = *FnIdOrErr;
1012 else {
1013 // Negotiation failed. Notify the handler then return the negotiate-failed
1014 // error.
1015 cantFail(Handler(make_error<ResponseAbandoned>()));
1016 return FnIdOrErr.takeError();
1017 }
1018
1019 SequenceNumberT SeqNo; // initialized in locked scope below.
1020 {
1021 // Lock the pending responses map and sequence number manager.
1022 std::lock_guard<std::mutex> Lock(ResponsesMutex);
1023
1024 // Allocate a sequence number.
1025 SeqNo = SequenceNumberMgr.getSequenceNumber();
1026 assert(!PendingResponses.count(SeqNo) &&
1027 "Sequence number already allocated");
1028
1029 // Install the user handler.
1030 PendingResponses[SeqNo] =
1031 detail::createResponseHandler<ChannelT, typename Func::ReturnType>(
1032 std::move(Handler));
1033 }
1034
1035 // Open the function call message.
1036 if (auto Err = C.startSendMessage(FnId, SeqNo)) {
1037 abandonPendingResponses();
1038 return Err;
1039 }
1040
1041 // Serialize the call arguments.
1042 if (auto Err = detail::HandlerTraits<typename Func::Type>::serializeArgs(
1043 C, Args...)) {
1044 abandonPendingResponses();
1045 return Err;
1046 }
1047
1048 // Close the function call messagee.
1049 if (auto Err = C.endSendMessage()) {
1050 abandonPendingResponses();
1051 return Err;
1052 }
1053
1054 return Error::success();
1055 }

1173 Handler PRHandler;

1186 Lock.unlock();

1189 SeqNo);

1194 "If we didn't find a response handler we should have bailed out");
1195
1196 if (auto Err = PRHandler->handleResponse(C)) {
1197 abandonPendingResponses();
1198 return Err;

1211 // Find the remote FunctionId for the given function.
1212 template <typename Func>
1213 Expected<FunctionIdT> getRemoteFunctionId(bool NegotiateIfNotInMap,
1214 bool NegotiateIfInvalid) {
1215 bool DoNegotiate;
1216
1217 // Check if we already have a function id...
1218 auto I = RemoteFunctionIds.find(Func::getPrototype());
1219 if (I != RemoteFunctionIds.end()) {
1220 // If it's valid there's nothing left to do.
1221 if (I->second != getInvalidFunctionId())
1222 return I->second;
1223 DoNegotiate = NegotiateIfInvalid;
1224 } else
1225 DoNegotiate = NegotiateIfNotInMap;
1226
1227 // We don't have a function id for Func yet, but we're allowed to try to
1228 // negotiate one.
1229 if (DoNegotiate) {
1230 auto &Impl = static_cast<ImplT &>(*this);
1231 if (auto RemoteIdOrErr =
1232 Impl.template callB<OrcRPCNegotiate>(Func::getPrototype())) {
1233 RemoteFunctionIds[Func::getPrototype()] = *RemoteIdOrErr;
1234 if (*RemoteIdOrErr == getInvalidFunctionId())
1235 return make_error<CouldNotNegotiate>(Func::getPrototype());
1236 return *RemoteIdOrErr;
1237 } else
1238 return RemoteIdOrErr.takeError();
1239 }
1240
1241 // No key was available in the map and we weren't allowed to try to
1242 // negotiate one, so return an unknown function error.
1243 return make_error<CouldNotNegotiate>(Func::getPrototype());
1244 }

1554/// Allows a set of asynchrounous calls to be dispatched, and then
1555/// waited on as a group.
1556class ParallelCallGroup {
1557public:

1562 /// Make as asynchronous call.
1563 template <typename AsyncDispatcher, typename HandlerT, typename... ArgTs>
1564 Error call(const AsyncDispatcher &AsyncDispatch, HandlerT Handler,
1565 const ArgTs &...Args) {
1566 // Increment the count of outstanding calls. This has to happen before
1567 // we invoke the call, as the handler may (depending on scheduling)
1568 // be run immediately on another thread, and we don't want the decrement
1569 // in the wrapped handler below to run before the increment.
1570 {
1571 std::unique_lock<std::mutex> Lock(M);
1572 ++NumOutstandingCalls;
1573 }
1574
1575 // Wrap the user handler in a lambda that will decrement the
1576 // outstanding calls count, then poke the condition variable.
1577 using ArgType = typename detail::ResponseHandlerArg<
1578 typename detail::HandlerTraits<HandlerT>::Type>::ArgType;
1579 auto WrappedHandler = [this, Handler = std::move(Handler)](ArgType Arg) {
1580 auto Err = Handler(std::move(Arg));
1581 std::unique_lock<std::mutex> Lock(M);
1582 --NumOutstandingCalls;
1583 CV.notify_all();
1584 return Err;
1585 };
1586
1587 return AsyncDispatch(std::move(WrappedHandler), Args...);
1588 }
1589
1590 /// Blocks until all calls have been completed and their return value
1591 /// handlers run.
1592 void wait() {
1593 std::unique_lock<std::mutex> Lock(M);
1594 while (NumOutstandingCalls > 0)
1595 CV.wait(Lock);
1596 }
1597
1598private:
1599 std::mutex M;
1600 std::condition_variable CV;
1601 uint32_t NumOutstandingCalls = 0;
1602};
