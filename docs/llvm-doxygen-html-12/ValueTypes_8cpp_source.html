148/// getEVTString - This function returns value type as a string, e.g. "i32".
149std::string EVT::getEVTString() const {
150 switch (V.SimpleTy) {
151 default:
152 if (isVector())
153 return (isScalableVector() ? "nxv" : "v") +
154 utostr(getVectorElementCount().getKnownMinValue()) +
155 getVectorElementType().getEVTString();
156 if (isInteger())
157 return "i" + utostr(getSizeInBits());
158 if (isFloatingPoint())
159 return "f" + utostr(getSizeInBits());
160 llvm_unreachable("Invalid EVT!");
161 case MVT::bf16: return "bf16";
162 case MVT::ppcf128: return "ppcf128";
163 case MVT::isVoid: return "isVoid";
164 case MVT::Other: return "ch";
165 case MVT::Glue: return "glue";
166 case MVT::x86mmx: return "x86mmx";
167 case MVT::x86amx: return "x86amx";
168 case MVT::Metadata: return "Metadata";
169 case MVT::Untyped: return "Untyped";
170 case MVT::funcref: return "funcref";
171 case MVT::externref: return "externref";
172 }
173}

487/// Return the value type corresponding to the specified type. This returns all
488/// pointers as MVT::iPTR. If HandleUnknown is true, unknown types are returned
489/// as Other, otherwise they are invalid.
490MVT MVT::getVT(Type *Ty, bool HandleUnknown){
491 switch (Ty->getTypeID()) {
492 default:
493 if (HandleUnknown) return MVT(MVT::Other);
494 llvm_unreachable("Unknown type!");
495 case Type::VoidTyID:
496 return MVT::isVoid;
497 case Type::IntegerTyID:
498 return getIntegerVT(cast<IntegerType>(Ty)->getBitWidth());
499 case Type::HalfTyID: return MVT(MVT::f16);
500 case Type::BFloatTyID: return MVT(MVT::bf16);
501 case Type::FloatTyID: return MVT(MVT::f32);
502 case Type::DoubleTyID: return MVT(MVT::f64);
503 case Type::X86_FP80TyID: return MVT(MVT::f80);
504 case Type::X86_MMXTyID: return MVT(MVT::x86mmx);
505 case Type::X86_AMXTyID: return MVT(MVT::x86amx);
506 case Type::FP128TyID: return MVT(MVT::f128);
507 case Type::PPC_FP128TyID: return MVT(MVT::ppcf128);
508 case Type::PointerTyID: return MVT(MVT::iPTR);
509 case Type::FixedVectorTyID:
510 case Type::ScalableVectorTyID: {
511 VectorType *VTy = cast<VectorType>(Ty);
512 return getVectorVT(
513 getVT(VTy->getElementType(), /*HandleUnknown=*/ false),
514 VTy->getElementCount());
515 }
516 }
517}

519/// getEVT - Return the value type corresponding to the specified type. This
520/// returns all pointers as MVT::iPTR. If HandleUnknown is true, unknown types
521/// are returned as Other, otherwise they are invalid.
522EVT EVT::getEVT(Type *Ty, bool HandleUnknown){
523 switch (Ty->getTypeID()) {
524 default:
525 return MVT::getVT(Ty, HandleUnknown);
526 case Type::IntegerTyID:
527 return getIntegerVT(Ty->getContext(), cast<IntegerType>(Ty)->getBitWidth());
528 case Type::FixedVectorTyID:
529 case Type::ScalableVectorTyID: {
530 VectorType *VTy = cast<VectorType>(Ty);
531 return getVectorVT(Ty->getContext(),
532 getEVT(VTy->getElementType(), /*HandleUnknown=*/ false),
533 VTy->getElementCount());
534 }
535 }
